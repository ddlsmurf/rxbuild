 22 if (!RXBuild)
 23 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 24 	var RXBuild = { };
 25 if (!RXBuild.UI)
 26 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 27 	 RXBuild.UI = {};
 28 
 29 (function() {
 30 
 31 	var quickref_url = "./syntax_reference.xml";
 32 
 33 /** 
 34 	Creates a new instance of RXBuild.UI.QuickRefViewer
 35 	@class The RXBuild.UI.QuickRefViewer displays a quick reference in a user-friendly format
 36 	@property {HTMLElement} container The element that will hold the reference viewer
 37 	@property {String} id The unique name of this viewer
 38 	@constructor
 39 	@param {HTMLElement} container The element that will hold the reference viewer
 40 	@param {String} id The unique name of this viewer
 41 */
 42 RXBuild.UI.QuickRefViewer = function (container, id, className) {
 43 	this.container = container;
 44 	this.id = id;
 45 	this.div = document.createElement("DIV");
 46 	this.div.className = ((typeof(className) == "string") ? className + " " : "") + " quickRefPanel";
 47 	this.div.id = id;
 48 	this.xml = null;
 49 	container.appendChild(this.div);
 50 	this.__load();
 51 };
 52 RXBuild.UI.QuickRefViewer.prototype.constructor = RXBuild.UI.QuickRefViewer;
 53 
 54 /** Called with the new XML reference to display.
 55 	@param {XMLDocument} xml The xml version of the quickreference to display to the user
 56 */
 57 RXBuild.UI.QuickRefViewer.prototype.initXML = function(xml) {
 58 	function buildCategoryTab(category) {
 59 		var sName = category.getAttribute("name");
 60 		var sHtmlResult = [];
 61 		sHtmlResult.push("<table border=\"1\" cellPadding=\"2\" cellSpacing=\"0\">");
 62 		sHtmlResult.push("<tr><th>Example</th><th>Description</th>");
 63 		var oItems = category.getElementsByTagName("item");
 64 		for (var i=0; i < oItems.length; i++) {
 65 			sHtmlResult.push("<tr><td>");
 66 			sHtmlResult.push(RXBuild.Utils.getXMLInnerText(oItems[i].getElementsByTagName("sample")));
 67 			sHtmlResult.push("</td><td>");
 68 			sHtmlResult.push(RXBuild.Utils.getXMLInnerText(oItems[i].getElementsByTagName("description")));
 69 			sHtmlResult.push("</td></tr>");
 70 		};
 71 		sHtmlResult.push("</table>");
 72 		return new YAHOO.widget.Tab({
 73 			label: sName,
 74 			content: sHtmlResult.join("")
 75 		});
 76 	}
 77 	function buildDialectsTab(xml, id) {
 78 		var oDialects = xml.getElementsByTagName("dialect");
 79 		var sHtmlResult = [];
 80 		for (var i=0; i < oDialects.length; i++) {
 81 			if (i > 0) sHtmlResult.push("<hr />");
 82 			var oLang = oDialects[i];
 83 			sHtmlResult.push("<h2>");
 84 			sHtmlResult.push(RXBuild.Utils.getXMLInnerText(oLang.getElementsByTagName("name")).trim());
 85 			sHtmlResult.push("</h2>");
 86 			var oDescription = oLang.getElementsByTagName("description");
 87 			if (oDescription.length >= 1) {
 88 				sHtmlResult.push("<h3>Description</h3>");
 89 				sHtmlResult.push(RXBuild.Utils.getXMLInnerText(oDescription[0]).trim());
 90 			}
 91 			var oSource = oLang.getElementsByTagName("source");
 92 			if (oSource.length >= 1) {
 93 				sHtmlResult.push("<h3>Copyright notice</h3>");
 94 				sHtmlResult.push(RXBuild.Utils.getXMLInnerText(oSource).trim());
 95 			}
 96 		};
 97 		return new YAHOO.widget.Tab({
 98 			id: id,
 99 			label: "Dialects",
100 			content: sHtmlResult.join(""),
101 			active: true
102 		});
103 	}
104 	
105 	this.xml = xml;
106 	this.div.innerHTML = "";
107 	this.tabView = new YAHOO.widget.TabView({id: this.id + "_tabView"});
108 	
109 	var oCategories = xml.getElementsByTagName("category");
110 	for (var j=0; j < oCategories.length; j++) {
111 		this.tabView.addTab(buildCategoryTab(oCategories[j]));
112 	};
113 	
114 	this.tabView.addTab(buildDialectsTab(xml, this.id + "_dialects_index"));
115 	this.tabView.appendTo(this.div);
116 };
117 
118 /** Attempts to load the XML content and initialise the control
119 	@private
120 */
121 RXBuild.UI.QuickRefViewer.prototype.__load = function() {
122 	var oCallBack = {
123 		success: function(o) {
124 			this.initXML(o.responseXML);
125 		},  
126 		failure: function(o) {
127 			this.div.innerHTML = "Failed to retreive quick reference - " + o.statusText + "<br />";
128 			var oRetryLink = document.createElement("A");
129 			oRetryLink.href = "#";
130 			oRetryLink.innerHTML = "Retry";
131 			YAHOO.util.Event.addListener(oRetryLink, "click", RXBuild.Utils.createDelegate(this,
132 				RXBuild.UI.QuickRefViewer.prototype.__load));
133 			this.div.appendChild(oRetryLink);
134 		},  
135 		scope: this
136 	};
137 	this.div.innerHTML = "Loading ...";
138 	YAHOO.util.Connect.asyncRequest('GET', quickref_url, oCallBack, null);
139 };
140 
141 })();
