 19 /**	@fileOverview Contains the component that create and update views of regexp (the regexp test matcher is one such view)
 20 	@requires utils.js
 21 	@requires regex_controller.js
 22 	@requires regex_engine.js
 23 */
 24 
 25 if (!RXBuild)
 26 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 27 	var RXBuild = { };
 28 if (!RXBuild.UI)
 29 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 30 	 RXBuild.UI = {};
 31 
 32 (function() {
 33 	/** 
 34 		Creates a new instance of RegexViewer
 35 		@class The RegexViewer holds selectable, self-updating views of the regular expression
 36 		@property {HTMLElement} container The DOM element that will be the happy new parent of this control
 37 		@constructor
 38 		@param {HTMLElement} container The DOM element that should be the new parent of the tester control.
 39 		@param {String} id A unique name for this controls dom element
 40 		@param {String} className Optional. A CSS class name to apply to the container DIV. 
 41 	*/
 42 	RXBuild.UI.RegexViewer = function (container, id, header, editor, viewers) {
 43 		this.container = container;
 44 		this.editor = editor;
 45 		
 46 		this.viewers = viewers;
 47 		for (var i=0; i < this.viewers.length; i++) {
 48 			var oCurView = this.viewers[i];
 49 			oCurView[1] = new oCurView[1](this.container, this.id + "_" + oCurView[2]);
 50 		}
 51 		
 52 		var oAcceptableViewersMenu = [];
 53 		for (var i=0; i < this.viewers.length; i++) {
 54 			oAcceptableViewersMenu.push({text: this.viewers[i][0], value: i});
 55 		}
 56 		 
 57 		this.btnChangeView = new YAHOO.widget.Button({   
 58 		                        id: id + "_btnSelectView",   
 59 		                        label: "<em class=\"yui-button-label\">Results - tree</em>",
 60 		                        type: "menu",    
 61 		                        menu: oAcceptableViewersMenu,
 62 								menualignment: ["tl", "tl"],
 63 		                        container: header});
 64 		this.btnChangeView.on("selectedMenuItemChange", RXBuild.Utils.createDelegate(this, function (event) {  
 65 		    var oMenuItem = event.newValue;
 66 			this.setCurrentView(oMenuItem.value);
 67 		}));
 68 		this.btnAutoRefresh = new YAHOO.widget.Button({ type: "checkbox", label: "Auto-refresh", container:header, checked: true,
 69 			onclick: {
 70 				fn: function() { this.autoRefresh(); this.btnRefresh.set("disabled", this.btnAutoRefresh.get("checked")); },
 71 				scope: this
 72 			}});
 73 		this.btnRefresh = new YAHOO.widget.Button({ type: "push", label: "Refresh", container:header, disabled: true,
 74 			onclick: {
 75 				fn: function() { this.update(); },
 76 				scope: this
 77 			}});
 78 		
 79 		this.editor.onRegExpInvalidated.subscribe(RXBuild.Utils.createDelegate(this, function() { this.invalidate(); }));
 80 		this.editor.onRegExpApplyChanges.subscribe(RXBuild.Utils.createDelegate(this, function() { this.autoRefresh(); }));
 81 		
 82 		this.setCurrentView(0);
 83 	};
 84 	RXBuild.UI.RegexViewer.prototype.constructor = RXBuild.UI.RegexViewer;
 85 	
 86 	/** Refreshes the view only if auto-refresh is active
 87 	*/
 88 	RXBuild.UI.RegexViewer.prototype.autoRefresh = function() {
 89 		if (this.btnAutoRefresh.get("checked"))
 90 			this.update();
 91 	};
 92 	
 93 	/** Sets the currently active viewer to the one with the specified index
 94 		@param {Number} viewIndex The index in this instances viewers property of the view to display
 95 	*/
 96 	RXBuild.UI.RegexViewer.prototype.setCurrentView = function(viewIndex) {
 97 		var oView = this.viewers[viewIndex];
 98 		this.btnChangeView.set("label", ("<em class=\"yui-button-label\">" + oView[0] + "</em>"));
 99     for (var i=0; i < this.viewers.length; i++) {
100     	this.viewers[i][1].container.style.display = i == viewIndex ? "" : "none";
101     }
102 		if (this.activeViewer && this.activeViewer.hide) this.activeViewer.hide();
103 		this.activeViewer = oView[1];
104 		this.invalidate();
105 		this.autoRefresh();
106 	};
107 
108 	/** Runs the provided regular expression and updates the results view accordingly
109 	*/
110 	RXBuild.UI.RegexViewer.prototype.update = function() {
111 		this.activeViewer.updateRegExp(this.editor.getRX(), this.editor.getInput());
112 		YAHOO.util.Dom.setStyle(this.container, "opacity", "1.0");
113 	};
114 	/** Notifies the user that the display is no longer uptodate with the regular expression
115 	*/
116 	RXBuild.UI.RegexViewer.prototype.invalidate = function(regexp, inputText) {
117 		YAHOO.util.Dom.setStyle(this.container, "opacity", "0.3");
118 	};
119 })();
