 19 /**	@fileOverview Contains the component that matches the regexp agains sample input and displays a tree view of the matches
 20 	@requires utils.js
 21 	@requires regex_controller.js
 22 	@requires regex_engine.js
 23 */
 24 
 25 if (!RXBuild)
 26 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 27 	var RXBuild = { };
 28 if (!RXBuild.UI)
 29 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 30 	 RXBuild.UI = {};
 31 
 32 (function() {	
 33 	/** 
 34 		Creates a new instance of RXBuild.UI.HtmlResultTreeViewer (which inherits RXBuild.Engine.ResultListener)
 35 		@class The RXBuild.UI.HtmlResultTreeViewer displays the results of matching a regular expression inline with the text
 36 		@property {HTMLElement} container The DOM element that holds the results for display		
 37 		@base RXBuild.Engine.ResultListener
 38 		@constructor
 39 		@param {} container
 40 	*/
 41 	RXBuild.UI.HtmlResultTreeViewer = function (container, id) {
 42 		RXBuild.Engine.ResultListener.call(this);
 43 		this.container = document.createElement("DIV");
 44 		this.container.setAttribute("id", id);
 45 		container.appendChild(this.container);
 46 		this.tree = new YAHOO.widget.TreeView(this.container);
 47 		this.tree.subscribe("labelClick", RXBuild.UI.HtmlResultTreeViewer.prototype.clickedOnNode);
 48 	};
 49 	RXBuild.UI.HtmlResultTreeViewer.prototype = new RXBuild.Engine.ResultListener;
 50 	RXBuild.UI.HtmlResultTreeViewer.prototype.constructor = RXBuild.UI.HtmlResultTreeViewer;
 51 	/** Updates the view to reflect the provided regular expression and input
 52 		@param {RXBuild.RegExp} regex The new pattern
 53 		@param {String} input The sample input text in which to run the matches
 54 	*/
 55 	RXBuild.UI.HtmlResultTreeViewer.prototype.updateRegExp = function (regex, input) {
 56 		var oEngine = new RXBuild.Engine.BrowserEngine();
 57 		oEngine.runMatch(regex, input, this);
 58 	};
 59 	/** Event handler for the yahoo tree labelClick event */
 60 	RXBuild.UI.HtmlResultTreeViewer.prototype.clickedOnNode = function (node) {
 61 		function showNodeDetails(node) {
 62 			if ((typeof node.data) == "string")
 63 			{
 64 				RXBuild.UI.dlg.show("Match details", node.data);
 65 				return;
 66 			}
 67 			RXBuild.UI.dlg.show("Match details", node.data[2]);
 68 		}
 69 		if ((typeof node.data) == "object")
 70 		{
 71 			if (!node.hasChildren(false))
 72 				showNodeDetails(node);
 73 			else
 74 				node.toggle();
 75 		}
 76 		else
 77 			showNodeDetails(node);
 78 		return false;
 79 	};
 80 	/** Called when a new match session has started
 81 		@param {RXBuild.RegExp} regexp The pattern that will be ran
 82 		@param {String} inputText The input text against which to match
 83 	*/
 84 	RXBuild.UI.HtmlResultTreeViewer.prototype.startMatch = function(regexp, inputText) {
 85 		this.tree.removeChildren(this.tree.getRoot());
 86 		this._iLastIndex = 0;
 87 	};
 88 	/** Called once for each match.
 89 		@param {RXBuild.RegExp} regexp The pattern that is running
 90 		@param {String} inputText The input text against which to match
 91 		@param {Number} matchIndex The 0-based index of this match
 92 		@param {object} matchItem A single match description
 93 		@return {Boolean} True to continue matching, false to abort.
 94 	*/
 95 	RXBuild.UI.HtmlResultTreeViewer.prototype.foundMatch = function(regexp, inputText, matchIndex, matchItem) {
 96 		function getAbbridged(s) {
 97 			if (s == null)
 98 				return "<null>";
 99 			if (s.length <= 123)
100 				return s;
101 			return s.substr(0, 120) + "...";
102 		}
103 		
104   		var sAbb = getAbbridged(matchItem[0]).escapeHTML();
105 		var tmpNode = new YAHOO.widget.TextNode("Match  #" + (matchIndex + 1) + " @" + matchItem.index + " \"" + sAbb + "\"", this.tree.getRoot(), matchItem.length > 1);
106 		tmpNode.data = new Array(matchItem.index, matchItem[0].length, matchItem[0]);
107 		if (sAbb != matchItem[0] && matchItem.length != 1)
108 				(new YAHOO.widget.TextNode("<em>View full match</em>", tmpNode, false)).data = tmpNode.data;
109 		for (var iSubMatch = 1; iSubMatch < matchItem.length; iSubMatch++)
110 			(new YAHOO.widget.TextNode("#" + iSubMatch + ": \'" + getAbbridged(matchItem[iSubMatch]).escapeHTML() + "\'", tmpNode, false)).data = matchItem[iSubMatch];
111 
112 		return matchIndex < 99;
113 	};
114 	/** Called when matching is over.
115 		@param {RXBuild.RegExp} regexp The pattern that was run
116 		@param {String} inputText The input text against which to match
117 		@param {Integer} matchCount The number of matches (until aborting), or -1 if an error occured.
118 		@param {String} error If an error occured, the error object, otherwise, null
119 	*/
120 	RXBuild.UI.HtmlResultTreeViewer.prototype.finished = function(regexp, inputText, matchCount, error) {
121 		if (matchCount != -1) {
122 		  	if (matchCount == 0)
123 			  	new YAHOO.widget.HTMLNode("<strong><No Matches></strong>", this.tree.getRoot(), true, true);
124 		  	else if (matchCount == 99)
125 			  	new YAHOO.widget.HTMLNode("<strong>Stopped matching at 100 !!<strong/>", this.tree.getRoot(), true, true);
126 		} else {
127 			new YAHOO.widget.TextNode("Error in regexp pattern. " + error.message, this.tree.getRoot(), true);
128 		}
129 		this.tree.draw();
130 	};
131 })();
