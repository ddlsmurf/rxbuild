 19 /**	@fileOverview Holds the status display control
 20 	@requires utils.js
 21 */
 22 
 23 if (!RXBuild)
 24 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 25 	var RXBuild = { };
 26 if (!RXBuild.UI)
 27 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 28 	 RXBuild.UI = {};
 29 
 30 (function() {
 31 	/** 
 32 		Creates a new instance of RXBuild.UI.Status
 33 		@class The RXBuild.UI.Status controls an always-on-top div that shows a tasks status
 34 		@constructor
 35 		@param {String} divName The name of the div to use in the end
 36 	*/
 37 	RXBuild.UI.Status = function (divName) {
 38 		this.div = document.createElement("DIV");
 39 		this.div.setAttribute("class", "");
 40 		this.div.setAttribute("id", divName);
 41 		this.div.setAttribute("style", "position: absolute; top: 0px; z-index: 5; padding: 5px; margin: 5px; right: 30px; " +
 42 			"border: 1px solid white;");
 43 		this.div.style.display = "none";
 44 		this.div.style.backgroundColor = "#ffddaa";
 45 		YAHOO.util.Dom.setStyle(this.div, 'opacity', '0');
 46 		this.div.innerHTML = "Fo' shizzle";
 47 		document.body.appendChild(this.div);
 48 		this._oAutoHide = new RXBuild.UI.DelayedRefresh(5000);
 49 		this._isVisible = false;
 50 	};
 51 	RXBuild.UI.Status.prototype.constructor = RXBuild.UI.Status;
 52 	/** Displays the status in the top right
 53 		@param {String} text The text to display in the body
 54 		@param {String} color An (optional) color to use as the new background color (in format #fffff)
 55 		@param {Number} autohideTimeout The number of milliseconds after which to hide the status object
 56 	*/
 57 	RXBuild.UI.Status.prototype.show = function(text, color, autohideTimeout) {
 58 		this.div.innerHTML = text;
 59 		this._oAutoHide.setPendingEvent(null);
 60 		if (!this._isVisible || color) {
 61 			this._isVisible = true;
 62 			this.div.style.display = "";
 63 			var oAnim = { opacity: {to: 1}};
 64 			if (color)
 65 			  oAnim.backgroundColor= {to: color};
 66 			oAnim = new YAHOO.util.ColorAnim(this.div, oAnim, 1, YAHOO.util.Easing.easeOut);
 67 			oAnim.animate();
 68 		}
 69 		if (typeof(autohideTimeout) != "undefined")
 70 			this._oAutoHide.setPendingEvent(RXBuild.Utils.createDelegate(this, function() { this.hide(); }), autohideTimeout);
 71 	};
 72 	/** Hides this status from view
 73 	*/
 74 	RXBuild.UI.Status.prototype.hide = function() {
 75 		this._oAutoHide.setPendingEvent(null);
 76 		try {
 77 			this._isVisible = false;
 78 			var oAnim = new YAHOO.util.Anim(this.div, {
 79 				opacity: {to: 0}
 80 			}, 1, YAHOO.util.Easing.easeOut);
 81 			oAnim.onComplete.subscribe(RXBuild.Utils.createDelegate(this, function() {this.div.style.display = "none";})); 
 82 			oAnim.animate();		
 83 		} catch (e) {
 84 			this.div.style.display = "none";
 85 		}
 86 	};
 87 
 88 
 89 })();
