 21 /**
 22 	Returns a version of the string that is suitable for insertion in XML code
 23 	(& is replaced by &amp; and so on)
 24 	@type String
 25 */
 26 String.prototype.escapeXML = function () {
 27         return(
 28             this.replace(/&/g,'&')
 29                 .replace(/>/g,'>')
 30                 .replace(/</g,'<')
 31                 .replace(/'/g,''')
 32                 .replace(/"/g,'"')
 33         );
 34     };
 35 /**
 36 	Returns a version of the string that is has all standard 5 entities replaced
 37 	(&amp; is replaced by & and so on)
 38 	@type String
 39 */
 40 String.prototype.unescapeXML = function () {
 41         return(
 42             this
 43                 .replace(/>/g,'>')
 44                 .replace(/</g,'<')
 45                 .replace(/'/g,'\'')
 46                 .replace(/"/g,'\"')
 47 				.replace(/&/g,'&')
 48         );
 49     };

 51 /**
 52 	Returns a version of the string that is suitable for insertion in HTML code
 53 	@type String
 54 */
 55 String.prototype.escapeHTML = function () {
 56 	return this.escapeXML()
 57 		.replace(/\r\n/g, "<br />")
 58 		.replace(/\n/g, "<br />")
 59 		.replace(/\r/g, "<br />");
 60 };

 63 /**
 64 	Returns a version of the string that is suitable for insertion in C/JS type code (without quotes though)
 65 	@type String
 66 
 67 */
 68 String.prototype.escapeToBackslashes = function () {
 69        return(
 70            (this.replace(/\\/g,'\\\\')).
 71                replace(/'/g,'\\\'').
 72                replace(/"/g,'\\\"').
 73                replace(/\t/g,'\\t').
 74                replace(/\r/g,'\\r').
 75                replace(/\f/g,'\\f').
 76                replace(/\v/g,'\\v').
 77                replace(/\n/g,'\\n').
 78 			   replace(/[^a-zA-Z0-9_+*\/\\ .,?!@#$%^&*():;\[\]|><~-]/gi, function (s) { 
 79 				var sHexCode = (new Number(s.charCodeAt(0))).toString(16);
 80 				if (sHexCode.length == 1) sHexCode = "0" + sHexCode;
 81 				if (sHexCode.length == 2) return "\\x" + sHexCode;
 82 				if (sHexCode.length == 3) sHexCode = "0" + sHexCode;
 83 				if (sHexCode.length == 4) return "\\u" + sHexCode;
 84 				return "[Unexplainable char 0x" + sHexCode + "]"; })
 85        );
 86    };

 97 /**
 98 	Returns the number of identical characters in this string and the one in s
 99 	starting from the beginning.
100 	@param {String} s The other string to compare to
101 	@param {Number} startIndex Optional - The position from which to start
102 	@return {Number} The number of common characters between the two strings
103 */
104 String.prototype.findCommonPrefix = function (s, startIndex) {
105 	var i;
106 	if (typeof(startIndex) != "number") startIndex = 0;
107 	for (i = startIndex; i < this.length && i < s.length && this.charAt(i) == s.charAt(i); i++)
108 		;
109     return i - startIndex;
110 };

112 /**
113 	Returns the number of identical characters in this string and the one in s
114 	starting from the beginning using case insensitive .
115 	@param {String} s The other string to compare to
116 	@param {Number} startIndex Optional - The position from which to start
117 	@return {Number} The number of common characters between the two strings
118 */
119 String.prototype.findCommonPrefixCaseInsensitive = function (s, startIndex) {
120 	return this.toLocaleLowerCase().findCommonPrefix(s.toLocaleLowerCase());
121 };

178 /** Parses the current instance for url encoded key value pairs, in the form <code>key%20name=key%20value&key%20bool&key%20empty=</code>
179 	@return {Array of arrays of string} Returns an array with each element being a key and optional value pair.
180 */
181 String.prototype.parseURLKeyValues = function () {
182         var sItems = this.split(/&/);
183 		for (var i=0; i < sItems.length; i++) {
184 			var sPieces = sItems[i].split(/\=/);
185 			for (var j = sPieces.length - 1; j >= 0; j--){
186 				sPieces[j] = unescape(sPieces[j]);
187 			};
188 			sItems[i] = sPieces;
189 		}
190 		return sItems;
191     };

193 (function() {
194 	/** @private */
195 	function replaceValues(oldValues, newValues) {
196 		function findItemByName(name) {
197 			for (var j = oldValues.length - 1; j >= 0; j--){
198 				var oVal = oldValues[j];
199 				if (typeof(oVal) == 'string'){
200 					if (oVal == name) return j;
201 				} else if (oVal[0] == name) return j;
202 			};
203 			return -1;
204 		}
205 		function replaceItemByName(name, newValue) {
206 			var iItemToDel = findItemByName(name);
207 			if (iItemToDel > -1) oldValues[iItemToDel] = newValue;
208 			else oldValues.push(newValue);
209 		}
210 		function deleteItemByName(name) {
211 			var iItemToDel = findItemByName(name);
212 			if (iItemToDel > -1) oldValues.splice(iItemToDel, 1);
213 		}
214 		for (var i = newValues.length - 1; i >= 0; i--){
215 			var oItem = newValues[i];
216 			if (typeof(oItem) == "string")
217 				deleteItemByName(oItem);
218 			else
219 				replaceItemByName(oItem[0], oItem);
220 		};
221 	}
222 	/** @private */
223 	function encodeURLKeyValuePair(values) {
224 		var sResult = [];
225 		for (var i=0; i < values.length; i++) {
226 			var oItem = values[i];
227 			if (typeof(oItem) == "string")
228 				sResult.push(escape(oItem));
229 			else {
230 				var sItemString = [];
231 				for (var j=0; j < oItem.length; j++) {
232 					sItemString.push(escape(oItem[j]));
233 				};
234 				sResult.push(sItemString.join("="));
235 			}
236 		};
237 		return sResult.join("&");
238 	}
239 	
240 	/** Replaces the provide key value pairs, in the format as returned by parseURLKeyValues in the current instance
241 		and returns the resulting string.
242 		@param {Array of Array of string} newValues The values to edit in the current instance. The following formats are acceptable:
243 		<ul>
244 			<li><code>["abc"]</code> - to delete the key with name abc</li>
245 			<li><code>[["abc"]]</code> - to empty the key abc (but keep it present)</li>
246 			<li><code>[["abc", "def"]]</code> - to replace or add the key named abc with the value def</li>
247 		</ul>
248 		@return {String} A string encoded in the format <code>key%20name=key%20value&key%20bool&key%20empty=</code>
249 	*/
250 	String.prototype.replaceURLKeys = function(newValues) {
251 		var oOldValue = this.parseURLKeyValues();
252 		replaceValues(oOldValue, newValues);
253 		return encodeURLKeyValuePair(oOldValue);
254 	};

264 /** Tries really hard to build a flat array with all parameters
265 	If the first argument is an array, it will be the one to which the other items are added.
266 	You can call JoinArrays(null, ...) to make sure you get a new array. It returns null if the 
267 	result would have been empty.
268 	@param ... {Array} A list of arrays.
269 	@type Array
270 	@static
271 */
272 RXBuild.Utils.JoinArrays = function () {
273 	var oArg = Array.prototype.slice.call(arguments);
274 	var oResult = oArg[0];
275 	var i = 1;
276 	if (oResult != null && !(oResult instanceof Array))
277 	{
278 		i = 0;
279 		oResult = null;
280 	}
281 	if (oResult == null)
282 		oResult = [];
283 	for (; i < oArg.length; i++)
284 		{
285 			var oItem = oArg[i];
286 			if (oItem != null)
287 				if (oItem instanceof Array)
288 					for (var j=0; j < oItem.length; j++)
289 						RXBuild.Utils.JoinArrays(oResult, oItem[j]);
290 				else
291 					oResult.push(oItem);
292 		}
293 	if (oResult.length == 0)
294 		return null;
295 	return oResult;
296 };

308 /**
309 	Generates a recursive print out of the provided object (o), skipping any properties
310 	contained in skip, and functions (unless requested).
311 	@param {object} o The object to explore.
312 	@param {String[]} skip A list of names of properties to exclude from the result
313 	@param {boolean} includeFunctions If set to true, function names are printed out too
314 	@param {String} i Initial indent string. (you can set to null)
315 	@type String
316 	@static
317 */
318 RXBuild.Utils.getObjectDesc = function(o, skip, includeFunctions, i) {
319 	function arrayHasString(arr, txt) {
320 		for (var s in arr) if (arr[s] == txt) return true;
321 		return false;
322 	}
323 	if (i == null) i = "";
324 	if (o == null) return "<null>";
325 	var oType = typeof(o);
326 	if (oType == "string") return "'" + o.escapeToBackslashes() + "'";
327 	if (oType == "number" || oType == "boolean") return o;
328 	if (oType == "function") return "{...}";
329 	var sResult = oType + "\n" + i;
330 	var bFirst = true;
331 	for (var prop in o)
332 		if (o.hasOwnProperty(prop) &&
333 			(includeFunctions || typeof(o[prop]) != "function") &&
334 			(skip == null || !arrayHasString(skip, prop))) {
335 			sResult += (bFirst ? "" : "\n" + i) + "-" + prop + ": " + RXBuild.Utils.getObjectDesc(o[prop], skip, includeFunctions, i + "  |");
336 			if (bFirst) bFirst = false;
337 		}
338 	return sResult;
339 	};

341 /** Builds an optimised regular expression to parse a list of tokens.
342 	@param {String[]} tokens An array of tokens to match
343 	@param {Boolean} noncapturing If set to true, generated groups are non-capturing (ie: <em>(?:)</em>)
344 	@param {Boolean} caseinsensitive If set to true, tokens are considered case insensitive
345 	@return {String} The regular expression that would match those tokens
346 */
347 RXBuild.Utils.buildRegexpFromTokens = function (tokens, noncapturing, caseinsensitive) {
348 	var root = [];
349 	function addNode(parent, str) { //Very dumb implementation of a suffix tree (but its home made - so dont you mention no fancy log(n) mumbo =)
350 		for (var i = 0; i < parent.length; i++) {
351 			var iCommon = caseinsensitive ? str.findCommonPrefixCaseInsensitive(parent[i].s) : str.findCommonPrefix(parent[i].s);
352 			if (iCommon == str.length) {
353 				// Found my terminal node
354 				parent[i].terminal = true;
355 				return ;
356 			} else if (iCommon > 0) {
357 				if (iCommon == parent[i].s.length) {
358 					// Found a node I should be under
359 					if (!parent[i].children)
360 						parent[i].children = [];
361 					addNode(parent[i].children, str.substr(iCommon));
362 					return;
363 				} else {
364 					// Found a node I must split and get under
365 					var sCommonPart = str.substr(0, iCommon);
366 					var oNodeToSplit = parent[i];
367 					var oNewChildFromPrevious = {
368 						s: oNodeToSplit.s.substr(iCommon)
369 						};
370 					if (oNodeToSplit.terminal) oNewChildFromPrevious.terminal = true;
371 					if (oNodeToSplit.children) oNewChildFromPrevious.children = oNodeToSplit.children;
372 					var sSuffix = str.substr(iCommon);
373 					var oNewChild = {
374 						s: sCommonPart,
375 						children: [oNewChildFromPrevious]
376 						};
377 					parent[i] = oNewChild;
378 					if (sSuffix.length == 0)
379 						oNewChild.terminal = true;
380 					else
381 						addNode(oNewChild.children, sSuffix);
382 					return;
383 				}
384 			}
385 		}
386 		// Couldnt find a place for myself, so made myself cosy
387 		parent.push({s: str, terminal: true});
388 	}
389 	function buildRegExp(parent, res, noncapturing) {
390 		for (var i=0; i < parent.length; i++) {
391 			if (i > 0) res.push("|");
392 			if (parent[i].children) {
393 				res.push(parent[i].s.escapeRegexp());
394 				res.push(noncapturing ? "(?:" : "(");
395 				buildRegExp(parent[i].children, res, noncapturing);
396 				res.push(")");
397 			} else {
398 				if (!parent[i].terminal) throw ("ASSERT: (This should never happen) Node is not terminal and has no children: '" + parent[i].s + "'");
399 				res.push(parent[i].s.escapeRegexp());
400 			}
401 			if (parent[i].children && parent[i].terminal) {
402 				res.push("?");
403 			}
404 		};
405 
406 	}
407 	tokens = tokens.sort();
408 	for (var i = tokens.length - 1; i >= 0; i--){
409 		if (tokens[i].trim() == "")
410 			tokens.pop();
411 		else
412 			tokens[i] = tokens[i].trim();
413 	};
414 	for (var j=0; j < tokens.length; j++)
415 		addNode(root, tokens[j]);
416 	var oRes = [];
417 	buildRegExp(root, oRes, noncapturing);
418 	return "\\b" + oRes.join("") + "\\b";
419 };
