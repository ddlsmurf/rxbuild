 31 /** @class The LiteralMatcher class matches the provided text literally
 32 	@property {String} texttomatch The text to look for in the input stream
 33 	@base RXBuild.Dom.Node
 34 	@constructor
 35 	@param {String} textToMatch The text to look for in the input stream
 36 */
 37 RXBuild.Dom.LiteralMatcher = function (textToMatch)
 38 {
 39 	RXBuild.Dom.Node.call(this);
 40 	this.texttomatch = textToMatch;	
 41 };
 42 RXBuild.Dom.LiteralMatcher.prototype = new RXBuild.Dom.Node;
 43 RXBuild.Dom.LiteralMatcher.prototype.constructor = RXBuild.Dom.LiteralMatcher;
 44 
 45 /** Helper method allowing the DOM to transform itself by concatenating all consecutive literal matchers
 46 	@private
 47 	@return {RXBuild.Dom.Node} A reference to the node that should replace this instance
 48 */
 49 RXBuild.Dom.LiteralMatcher.prototype.Flatten = function() {
 50 	while (this.next && (this.next instanceof RXBuild.Dom.LiteralMatcher))
 51 	{
 52 		this.texttomatch += this.next.texttomatch;
 53 		this.AddTokens(this.next.tokens);
 54 		this.next = this.next.next;
 55 	}
 56 	return RXBuild.Dom.Node.prototype.Flatten.call(this);
 57 };

 59 /** Returns a human readable description of this match
 60 	@return {String} The description of this node in human readable format.
 61 */
 62 RXBuild.Dom.LiteralMatcher.prototype.GetDescription = function()
 63 {
 64 	return "[LitMatch '" + this.texttomatch + "']";
 65 };

 67 /** Returns an HTML description of this match

 70 RXBuild.Dom.LiteralMatcher.prototype.GetHtml = function() {
 71 	return "Matches <code>" + this.texttomatch.escapeToBackslashes().escapeHTML() + "</code>";

 74 /** @class The PositionalMatch class matches the specific positions (such as $, ^, \b and \B)
 75 @property {String} posChar The character representing the position to match
 76 @property {Boolean} multiline If set, the regexp was compiled in multiline
 77 @base RXBuild.Dom.Node
 78 @constructor
 79 @param {String} positionalChar The character representing the position to match
 80 @param {Boolean} multiline Whether the regexp was compiled in multiline
 81 */
 82 RXBuild.Dom.PositionalMatch = function (positionalChar, multiline)
 83 {
 84 	RXBuild.Dom.Node.call(this);
 85 	this.posChar = positionalChar;
 86 	if (multiline) this.multiline = multiline;
 87 };
 88 RXBuild.Dom.PositionalMatch.prototype = new RXBuild.Dom.Node;
 89 RXBuild.Dom.PositionalMatch.prototype.constructor = RXBuild.Dom.PositionalMatch;
 90 
 91 /** Returns a human readable description of this match
 92 	@return {String} The description of this node in human readable format.
 93 */
 94 RXBuild.Dom.PositionalMatch.prototype.GetDescription = function() {
 95 	return "[Positional " + this.posChar + "]";
 96 };
 97 
 98 /** Returns an HTML description of this match
 99 	@return {String} An HTML description of this match
100 */
101 RXBuild.Dom.PositionalMatch.prototype.GetHtml = function() {
102 	if (this.posChar == "^")
103 		return "At the beginning of the string" + (this.multiline ? " (or line)" : "");
104 	if (this.posChar == "$")
105 		return "At the end of the string" + (this.multiline ? " (or line)" : "");
106 	if (this.posChar == "\\b")
107 		return "At a word boundary";
108 	if (this.posChar == "\\B")
109 		return "Not at a word boundary";
110 	RXBuild.Dom.Node.prototype.GetHtml.call(this);
111 };

113 /** @class The CharacterRangeMatch class matches a range of characters (eg: [a-z])
114 @property {Boolean} reversed If set, the range match is negated
115 @property {Array} ranges An array of objects representing acceptable ranges
116 @base RXBuild.Dom.Node
117 @constructor
118 @param {Boolean} reversed If set, the range is negated
119 */
120 RXBuild.Dom.CharacterRangeMatch = function (reversed)
121 {
122 	RXBuild.Dom.Node.call(this);
123 	this.ranges = new Array();
124 	this.pending = new Array();
125 	this.reversed = reversed;
126 	this.include = "";
127 	this.exclude = "";
128 	this.asString = "";
129 };
130 RXBuild.Dom.CharacterRangeMatch.prototype = new RXBuild.Dom.Node;
131 RXBuild.Dom.CharacterRangeMatch.prototype.constructor = RXBuild.Dom.CharacterRangeMatch;
132 RXBuild.Dom.CharacterRangeMatch.prototype.AddChar = function(text) {
133 	if (this.reversed)
134 		this.exclude += text;
135 	else
136 		this.include += text;
137 	this.asString +=
138 	(this.reversed ? 
139 		(this.asString != "" ? ", (but not " : "Anything (except ") : 
140 		(this.asString != "" ? ", or " : "")) +
141 	"<code>" + text.escapeHTML().escapeToBackslashes() + "</code>" + 
142 	(this.reversed ? ")" : "");
143 };
144 RXBuild.Dom.CharacterRangeMatch.prototype.AddChars = function(chars) {
145 	if (chars != null && chars instanceof RXBuild.Dom.LiteralMatcher)
146 	{
147 		this.AddTokens(chars.tokens);
148 		chars = chars.texttomatch;
149 	}
150   if (this.pending.length == 0 && chars == "^" && (this.tokens == null || this.tokens.length == 0)) {
151     this.reversed = true;
152     return;
153   }
154 	if (this.pending.length == 0 && chars == "-")
155 	{
156 		this.AddChar(chars);
157 		return;
158 	}
159 	if (this.pending.length == 1 && chars != "-")
160 			this.AddChar(this.pending.pop());
161 	if (this.pending.length == 2 && chars != null)
162 	{
163 		this.AddRange(this.pending[0], chars, false);
164 		this.pending = new Array();
165 		chars = null;
166 	}	
167 	while (this.pending.length > 0 && chars == null)
168 		this.AddChar(this.pending.pop());
169 	if (chars != null)
170 		this.pending.push(chars);
171 };
172 RXBuild.Dom.CharacterRangeMatch.prototype.AddClass = function(chars) {
173 	if (!chars || chars.length == 0) throw "Invalid char class \'" + chars + "\'";
174 	var bReversed = chars.charAt(0) == "^";
175 	if (bReversed) chars = chars.substr(1);
176 	var oPending = this.pending;
177 	this.pending = [];
178 	if (bReversed) this.reversed = !this.reversed;
179 
180 	for (var i=0; i < chars.length; i++) {
181 		this.AddChars(chars.charAt(i));
182 	};
183 	this.AddChars(null);
184 	
185 	if (bReversed) this.reversed = !this.reversed;
186 	this.pending = oPending;
187 };
188 RXBuild.Dom.CharacterRangeMatch.prototype.AddRange = function(start_range, stop_range, reverseRange) {
189 	if (start_range instanceof RXBuild.Dom.LiteralMatcher)
190 		start_range = start_range.texttomatch;
191 	if (stop_range instanceof RXBuild.Dom.LiteralMatcher)
192 		stop_range = stop_range.texttomatch;
193 	if (this.reversed) reverseRange = !reverseRange;
194 	this.ranges.push({
195 		exclude: reverseRange,
196 		from: start_range,
197 		to: stop_range
198 	});
199 	this.asString += 
200 		 (reverseRange ?
201 			(this.asString != "" ? ", (but not " : "Anything (except from ") :
202 			(this.asString != "" ? ", or from " : "Anything from ")) + 
203 		"<code>" + start_range.escapeHTML().escapeToBackslashes() + "</code>" +
204 		" to " +
205 		"<code>" + stop_range.escapeHTML().escapeToBackslashes() + "</code>" + (reverseRange ? ")" : "");
206 };

208 /** Helper method allowing the DOM to transform itself. Used to flush pending characters.
209 	@private
210 	@return {RXBuild.Dom.Node} A reference to the node that should replace this instance
211 */
212 RXBuild.Dom.CharacterRangeMatch.prototype.Flatten = function() {
213 	if (this.pending != null)
214 	{
215 		this.AddChars(null);
216 		this.pending = null;
217 	}
218 	return RXBuild.Dom.Node.prototype.Flatten.call(this);
219 };
220 
221 /** Returns a human readable description of this match
222 	@return {String} The description of this node in human readable format.
223 */
224 RXBuild.Dom.CharacterRangeMatch.prototype.GetDescription = function() {
225 	if (this.asString == "")
226 		if (this.reversed)
227 			return "[Match any single character]";
228 		else
229 			return "[Never match anything]";
230 	return "[RXBuild.Dom.CharacterRangeMatch " + this.asString + (this.reversed ? "(reversed)" : "") + "]";
231 };
232 
233 /** Returns an HTML description of this match
234 	@return {String} An HTML description of this match
235 */
236 RXBuild.Dom.CharacterRangeMatch.prototype.GetHtml = function() {
237 	if (this.asString == "")
238 		if (this.reversed)
239 			return "Match any single character";
240 		else
241 			return "Never match anything (empty character range)";
242 	return this.asString;
243 };
