 19 /**	@fileOverview Holds the status display control
 20 	@requires utils.js
 21 */
 22 /*jsl:define YAHOO
 23 */
 24 if (!RXBuild)
 25 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 26 	var RXBuild = { };
 27 if (!RXBuild.UI)
 28 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 29 	 RXBuild.UI = {};
 30 
 31 (function() {
 32 	/** 
 33 		Creates a new instance of RXBuild.UI.Status
 34 		@class The RXBuild.UI.Status controls an always-on-top div that shows a tasks status
 35 		@constructor
 36 		@param {String} divName The name of the div to use in the end
 37 		@param {String} className Optional. A CSS class name to apply to the container DIV. 
 38 	*/
 39 	RXBuild.UI.Status = function (divName, className) {
 40 		this.div = document.createElement("DIV");
 41 		this.div.setAttribute("class", (typeof (className) == "string" ? className : ""));
 42 		this.div.setAttribute("id", divName);
 43 		this.div.setAttribute("style", "position: absolute; top: 0px; z-index: 5; padding: 5px; margin: 5px; right: 30px; " +
 44 			"border: 1px solid white;");
 45 		this.div.style.display = "none";
 46 		this.div.style.backgroundColor = "#ffddaa";
 47 		YAHOO.util.Dom.setStyle(this.div, 'opacity', '0');
 48 		this.div.innerHTML = "Fo' shizzle";
 49 		document.body.appendChild(this.div);
 50 		this._oAutoHide = new RXBuild.UI.DelayedRefresh(5000);
 51 		this._isVisible = false;
 52 	};
 53 	RXBuild.UI.Status.prototype.constructor = RXBuild.UI.Status;
 54 	/** Displays the status in the top right
 55 		@param {String} text The text to display in the body
 56 		@param {String} color An (optional) color to use as the new background color (in format #fffff)
 57 		@param {Number} autohideTimeout The number of milliseconds after which to hide the status object
 58 	*/
 59 	RXBuild.UI.Status.prototype.show = function(text, color, autohideTimeout) {
 60 		this.div.innerHTML = text;
 61 		this._oAutoHide.setPendingEvent(null);
 62 		if (!this._isVisible || color) {
 63 			this._isVisible = true;
 64 			this.div.style.display = "";
 65 			var oAnim = { opacity: {to: 1}};
 66 			if (color)
 67 			  oAnim.backgroundColor= {to: color};
 68 			oAnim = new YAHOO.util.ColorAnim(this.div, oAnim, 1, YAHOO.util.Easing.easeOut);
 69 			oAnim.animate();
 70 		}
 71 		if (typeof(autohideTimeout) != "undefined")
 72 			this._oAutoHide.setPendingEvent(RXBuild.Utils.createDelegate(this, function() { this.hide(); }), autohideTimeout);
 73 	};
 74 	/** Hides this status from view
 75 	*/
 76 	RXBuild.UI.Status.prototype.hide = function() {
 77 		this._oAutoHide.setPendingEvent(null);
 78 		try {
 79 			this._isVisible = false;
 80 			var oAnim = new YAHOO.util.Anim(this.div, {
 81 				opacity: {to: 0}
 82 			}, 1, YAHOO.util.Easing.easeOut);
 83 			oAnim.onComplete.subscribe(RXBuild.Utils.createDelegate(this, function() {this.div.style.display = "none";})); 
 84 			oAnim.animate();		
 85 		} catch (e) {
 86 			this.div.style.display = "none";
 87 		}
 88 	};
 89 
 90 
 91 })();
