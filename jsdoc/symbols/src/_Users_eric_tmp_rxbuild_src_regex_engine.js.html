  9 if (!RXBuild.Engine)
 10 	/**
 11 		@namespace RXBuild.Engine
 12 		Holds a facade for running regular expressions.

 14 	RXBuild.Engine = { };

 16 (function() {
 17 
 18 
 19 	/** 
 20 		Creates a new instance of RXBuild.Engine.ResultListener
 21 		@class The RXBuild.Engine.ResultListener is an abstract interface that receives the result of a regular expression matcher.
 22 		@constructor
 23 	*/
 24 	RXBuild.Engine.ResultListener = function () {
 25 	};
 26 	RXBuild.Engine.ResultListener.prototype.constructor = RXBuild.Engine.ResultListener;
 27 	/** Called when a new match session has started
 28 		@param {RXBuild.RegExp} regexp The pattern that will be ran
 29 		@param {String} inputText The input text against which to match
 30 	*/
 31 	RXBuild.Engine.ResultListener.prototype.startMatch = function(regexp, inputText) {};
 32 	/** Called once for each match.
 33 		@param {RXBuild.RegExp} regexp The pattern that is running
 34 		@param {String} inputText The input text against which to match
 35 		@param {Number} matchIndex The 0-based index of this match
 36 		@param {object} matchItem A single match description
 37 		@return {Boolean} True to continue matching, false to abort.
 38 	*/
 39 	RXBuild.Engine.ResultListener.prototype.foundMatch = function(regexp, inputText, matchIndex, matchItem) {
 40 		return true;
 41 	};
 42 	/** Called when matching is over.
 43 		@param {RXBuild.RegExp} regexp The pattern that was run
 44 		@param {String} inputText The input text against which to match
 45 		@param {Integer} matchCount The number of matches (until aborting), or -1 if an error occured.
 46 		@param {String} error If an error occured, the error object, otherwise, null
 47 	*/
 48 	RXBuild.Engine.ResultListener.prototype.finished = function(regexp, inputText, matchCount, error) {};

 51 	/** 
 52 		Creates a new instance of RXBuild.Engine.BrowserEngine
 53 		@class The RXBuild.Engine.BrowserEngine A facade for using the current default (browser)
 54 		implementation of the regular expression
 55 		@property {String} name The name of this engine.
 56 		@constructor
 57 	*/
 58 	RXBuild.Engine.BrowserEngine = function () {
 59 	};
 60 	RXBuild.Engine.BrowserEngine.prototype.constructor = RXBuild.Engine.BrowserEngine;
 61 	RXBuild.Engine.BrowserEngine.prototype.name = "Safe - Default (browser's) implementation";
 62 	/** Runs the provided regular expression against the input text notifying listener of any matches
 63 		@param {RXBuild.RegExp} regexp The regular expression with the pattern to match
 64 		@param {String} input The input text being searched
 65 		@param {RXBuild.Engine.ResultListener} listener The object that will reveive the results of the matching
 66 	*/
 67 	RXBuild.Engine.BrowserEngine.prototype.runMatch = function(regexp, input, listener) {
 68 		listener.startMatch(regexp, input);
 69 		try
 70 		{
 71 			var oRegExp = new RegExp(regexp.pattern, regexp.options.str);
 72 		}
 73 		catch(err)
 74 		{
 75 			oRegExp = null;
 76 			listener.finished(regexp, input, -1, err);
 77 			return;
 78 		}
 79 		
 80 	  	var oMatches;
 81 	  	var i = 0;
 82 	  	while ((oMatches = oRegExp.exec(input)) != null)
 83 	  	{
 84 			if (!listener.foundMatch(regexp, input, i, oMatches))
 85 				break;
 86 			i++;
 87 			if (!regexp.options.g) break;
 88 	  	}
 89 		listener.finished(regexp, input, i, null);
 90 	};
 91 
 92 })();
