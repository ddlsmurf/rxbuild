 19 /**	@fileOverview This file holds a class that encapsulates a regular expression in kinda universal format
 20 	@requires utils.js
 21 	@requires regex_parser.js
 22 */
 23 
 24 if (!RXBuild)
 25 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 26 	var RXBuild = { };
 27 
 28 (function() {
 29 
 30 	/** 
 31 		Creates a new instance of RXBuild.RegExp
 32 		@class The RXBuild.RegExp class holds a regular expression as input by the user, and connects with the various operations
 33 		@property {String} pattern Gets the regular expression as input by the user
 34 		@property {Options} options Gets the options selected by the user. This object has the properties <strong>g</strong>, <strong>m</strong> and
 35 		 							<strong>i</strong> set to true if the corresponding option is active. There is also a <strong>str</strong> property
 36 									with the options as a string (eg.: "gi").
 37 		@constructor
 38 		@param {String} pattern The regular expression as set by the user
 39 		@param {Boolean} g True if the pattern is global
 40 		@param {Boolean} i True if the pattern is case insensitive
 41 		@param {Boolean} m True if the pattern is in multiline mode
 42 	*/
 43 	RXBuild.RegExp = function (pattern, g, i, m) {
 44 		this.pattern = pattern;
 45 		this.options = {};
 46 		if (g) this.options.g = true;
 47 		if (m) this.options.m = true;
 48 		if (i) this.options.i = true;
 49 		this.options.str = (i ? "i" : "") + (g ? "g" : "") + (m ? "m" : "");
 50 	};
 51 	RXBuild.RegExp.prototype.constructor = RXBuild.RegExp;
 52 	
 53 	/** @field {Array} Index of builder functions. It is an array of builder, each being an array of 2 items, the english name, and the function.
 54 	*/
 55 	RXBuild.RegExp.prototype.CodeBuilders = [];
 56 
 57 	/** Compiles the regexp using the custom parser, and returns the expressions DOM, or throws a string with errors encountered during parse
 58 		@return {Matcher} The root compiled matcher if the pattern was valid
 59 	*/
 60 	RXBuild.RegExp.prototype.compile = function()
 61 	{
 62 		var oParser = new regex_parser();
 63 		var oResult = oParser.compile(this.pattern, this.options.i, this.options.m, this.options.g);
 64 		if (typeof(oResult) != "string")
 65 			return oResult;
 66 		else
 67 			throw oResult;
 68 	};
 69 	/** Generates and returns an equivalent JavaScript code
 70 		@return {String} A snippet of JavaScript code that would run this regular expression.
 71 	*/
 72 	RXBuild.RegExp.prototype.getJSCode = function()
 73 	{
 74 		var sResult = "var oRegExp = new RegExp(" + this.pattern.escapeJS() + ", \"" + this.options.str + "\");\n";
 75 		sResult += "var oMatches;\n";
 76 		sResult += "while ((oMatches = oRegExp.exec(sSourceText_REPLACE_ME)) != null) {\n";
 77 		sResult += "    // Match in oMatches[0], Groups start at 1\n";
 78 		sResult += "}\n";
 79 		return sResult;
 80 	};
 81 	RXBuild.RegExp.prototype.CodeBuilders.push(["JavaScript", RXBuild.RegExp.prototype.getJSCode]);
 82 	
 83 	/** Generates and returns an equivalent C# code
 84 		@return {String} A snippet of C# code that would run this regular expression.
 85 	*/
 86 	RXBuild.RegExp.prototype.getCSCode = function()
 87 	{
 88 		var sResult = "System.Text.RegularExpressions.Regex oRegExp = new System.Text.RegularExpressions.Regex(" + this.pattern.escapeCS() + ", ";
 89 		sResult += "   RegexOptions.Compiled.RegexOptions.Compiled|RegexOptions.Compiled.RegexOptions.ECMAScript";
 90 		if (this.options.i) sResult += "|RegexOptions.Compiled.RegexOptions.IgnoreCase";
 91 		if (this.options.m) sResult += "|RegexOptions.Compiled.RegexOptions.Multiline";
 92 		sResult += ");\n";
 93 		sResult +=
 94 	    "foreach (Match oMatch in oRegExp.Matches(input))\n" +
 95 	    "{\n" +
 96 	    "    //Match in oMatch\n" +
 97 		"	 //Captures in oMatch.Groups[1+]\n" +
 98 	    "}\n";
 99 		return sResult;
100 	};
101 	RXBuild.RegExp.prototype.CodeBuilders.push(["C#", RXBuild.RegExp.prototype.getCSCode]);
102 	
103 	/** Generates and returns an equivalent Perl code
104 		@return {String} A snippet of Perl code that would run this regular expression.
105 	*/
106 	RXBuild.RegExp.prototype.getPerlCode = function() {
107 		var sRegExStart = "";
108 		var sRegex = this.pattern.normaliseNewLines();
109 		if (sRegex.indexOf("$") > -1) {
110 			var sHereDoc = "REGEX";
111 			var iOffset = 1;
112 			while (sRegex.indexOf("\n" + sHereDoc + "\n") > -1)
113 				sHereDoc = sHereDoc + "_" + (iOffset++);
114 			sRegExStart = "my $regex = <<'" + sHereDoc + "';\n" +
115 						  sRegex + "\n" +
116 						  sHereDoc + "\n" + 
117 						  "chop $regex;\n";
118 			sRegex = "$regex";
119 		}
120 		else
121 			sRegex = sRegex.replace(/\\/g,'\\\\').replace(/\//g,'\\/');
122 		return sRegExStart + "while ($ReplaceMeWithTheInputText =~ m/" + sRegex + "/" + this.options.str +") {\n" +
123 			"# matches are in $1, $2... position in pos $ReplaceMeWithTheInputText\n}"; 
124 	};
125 	RXBuild.RegExp.prototype.CodeBuilders.push(["Perl", RXBuild.RegExp.prototype.getPerlCode]);
126 	
127 	/** Generates and returns an equivalent Bash/Grep code
128 		@return {String} A snippet of Bash/Grep code that would run this regular expression.
129 	*/
130 	RXBuild.RegExp.prototype.getBashGrepCode = function () {
131 		var sRes = "grep -E -o ";
132 		var sOptions = "";
133 		if (this.options.i) sOptions += "-i ";
134 		if (!(this.options.g)) sOptions += "-m 1 ";
135 		if (!(this.options.m)) sRes = "# Warning - the multiline option is deactivated but not supported by grep (grep is line based).\n#It has been ignored. \n\n" + sRes;
136 		var sEscapedRegExp = this.pattern.replace(/\\/g,'\\\\').
137                replace(/\$/g,'\\\'').
138                replace(/"/g,'\\\"').
139                replace(/`/g,'\\`').
140                replace(/(\r(\n?))|\n/g,'\\\n');
141 		sEscapedRegExp = "-e \"" + sEscapedRegExp + "\"";
142 		sRes += sOptions + sEscapedRegExp;
143 		return sRes;
144 	};
145 	RXBuild.RegExp.prototype.CodeBuilders.push(["Bash and grep", RXBuild.RegExp.prototype.getBashGrepCode]);
146 
147 	/** Generates and return an equivalent Ruby code
148 	 *  @return {String} A snippet of ruby code that would run this regular expression.
149 	 */
150 	RXBuild.RegExp.prototype.getRubyCode = function () {
151 		var sRes = "";
152 		var sOptions = this.options.str.replace('g', '');
153 		var sRe = "/" + this.pattern.replace(/\//g, '\\/') + '/' + sOptions;
154 		if (this.options.g) {
155 			sRes += 'for match in replace_me_with_the_input_text.scan(' + sRe + ")\n";
156 			sRes += "  # Your matches are in match or match[n]\n";
157 			sRes += "end\n";
158 		} else {
159 			sRes += 'match_group = replace_me_with_the_input_text.match(' + sRe + ")\n";
160 			sRes += "# Your match are in match_group[n]\n";
161 		}
162 		return sRes;
163 	};
164 	RXBuild.RegExp.prototype.CodeBuilders.push(["Ruby", RXBuild.RegExp.prototype.getRubyCode]);
165 })();
