 19 /**	@fileOverview Contains the component that matches the regexp agains sample input and displays an inline result view of the matches
 20 	@requires utils.js
 21 	@requires regex_controller.js
 22 	@requires regex_engine.js
 23 */
 24 
 25 if (!RXBuild)
 26 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 27 	var RXBuild = { };
 28 if (!RXBuild.UI)
 29 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 30 	 RXBuild.UI = {};
 31 
 32 (function() {
 33 	/** 
 34 		Creates a new instance of RXBuild.UI.HtmlResultViewer (which inherits RXBuild.Engine.ResultListener)
 35 		@class The RXBuild.UI.HtmlResultViewer displays the results of matching a regular expression inline with the text
 36 		@property {HTMLElement} container The DOM element that holds the results for display		
 37 		@base RXBuild.Engine.ResultListener
 38 		@constructor
 39 		@param {} container
 40 	*/
 41 	RXBuild.UI.HtmlResultViewer = function (container, id) {
 42 		RXBuild.Engine.ResultListener.call(this);
 43 		this.container = document.createElement("DIV");
 44 		this.container.setAttribute("class", "rxbuild_regextester");
 45 		this.container.setAttribute("id", id);
 46 		container.appendChild(this.container);
 47 	};
 48 	RXBuild.UI.HtmlResultViewer.prototype = new RXBuild.Engine.ResultListener;
 49 	RXBuild.UI.HtmlResultViewer.prototype.constructor = RXBuild.UI.HtmlResultViewer;
 50 	/** Updates the view to reflect the provided regular expression and input
 51 		@param {RXBuild.RegExp} regex The new pattern
 52 		@param {String} input The sample input text in which to run the matches
 53 	*/
 54 	RXBuild.UI.HtmlResultViewer.prototype.updateRegExp = function (regex, input) {
 55 		var oEngine = new RXBuild.Engine.BrowserEngine();
 56 		oEngine.runMatch(regex, input, this);
 57 	};
 58 	
 59 	/** Called when a new match session has started
 60 		@param {RXBuild.RegExp} regexp The pattern that will be ran
 61 		@param {String} inputText The input text against which to match
 62 	*/
 63 	RXBuild.UI.HtmlResultViewer.prototype.startMatch = function(regexp, inputText) {
 64 		this.container.innerHTML = "";
 65 		this._iLastIndex = 0;
 66 	};
 67 	/** Called once for each match.
 68 		@param {RXBuild.RegExp} regexp The pattern that is running
 69 		@param {String} inputText The input text against which to match
 70 		@param {Number} matchIndex The 0-based index of this match
 71 		@param {object} matchItem A single match description
 72 		@return {Boolean} True to continue matching, false to abort.
 73 	*/
 74 	RXBuild.UI.HtmlResultViewer.prototype.foundMatch = function(regexp, inputText, matchIndex, matchItem) {
 75 		if (this._iLastIndex < matchItem.index)
 76 			this.container.appendChild(document.createTextNode(inputText.substr(this._iLastIndex, matchItem.index - this._iLastIndex)));
 77 		var oChild = document.createElement('SPAN');
 78 		oChild.className = "rxbuild_regexmatch";
 79 		oChild.innerHTML = matchItem[0].escapeHTML();
 80 		this.container.appendChild(oChild);
 81 		this._iLastIndex = matchItem.index + matchItem[0].length;
 82 		return true;
 83 	};
 84 	/** Called when matching is over.
 85 		@param {RXBuild.RegExp} regexp The pattern that was run
 86 		@param {String} inputText The input text against which to match
 87 		@param {Integer} matchCount The number of matches (until aborting), or -1 if an error occured.
 88 		@param {String} error If an error occured, the error object, otherwise, null
 89 	*/
 90 	RXBuild.UI.HtmlResultViewer.prototype.finished = function(regexp, inputText, matchCount, error) {
 91 		if (matchCount != -1) {
 92 		if (this._iLastIndex < inputText.length)
 93 			this.container.appendChild(document.createTextNode(inputText.substr(this._iLastIndex)));
 94 		} else {
 95 			var oChild = document.createElement('PRE');
 96 			oChild.className = "rxbuild_regexerror";
 97 			oChild.innerHTML = String(error.message).escapeHTML();
 98 			this.container.appendChild(oChild);
 99 		}
100 	};
101 })();
