 19 /** @fileOverview This file defines YUI dialog extensions
 20 	@requires utils.js
 21 */
 22 if (!RXBuild)
 23 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 24 	var RXBuild = { };
 25 if (!RXBuild.UI)
 26 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 27 	 RXBuild.UI = {};
 28 
 29 (function() {
 30 
 31 	/** 
 32 		Creates a new instance of TextAreaDialog
 33 		@class The RXBuild.UI.Dialog displays a YUI dialog with a text area or provided HTMLElement as main content, and configurable verbs
 34 		@property {String} divName The name of the div element that this item provided to YUI
 35 		@property {YAHOO.widget.Dialog} dialog The YUI dialog - used internally.
 36 		@property {YAHOO.widget.Button} _btnOk The YUI ok button - used internally.
 37 		@property {YAHOO.widget.Button} _btnCancel The YUI close/cancel button - used internally.
 38 		@property {HTMLTextArea} textBox The dynamically create text area - used internally.
 39 		@constructor
 40 		@param {String} divName The name of the div element that this item will provide to YUI
 41 	*/
 42 	RXBuild.UI.Dialog = function (divName) {
 43 		this.divName = divName;
 44 		this.dialog = new YAHOO.widget.Dialog(divName, 
 45 			{
 46 				width:"640px",
 47 				visible:false,
 48 				constraintoviewport:true,
 49 				effect:{effect:YAHOO.widget.ContainerEffect.FADE,duration:0.25},
 50 				fixedcenter: true,
 51 				hideaftersubmit: true, 
 52 				postmethod: "manual",
 53 				modal: true,
 54 				close: false,
 55 				buttons:  [
 56 					{
 57 						text:"Ok",
 58 						handler:
 59 							{
 60 								fn: function(e, btnId) { this._buttonPressed(btnId); },
 61 								obj: "ok",
 62 								scope: this
 63 							},
 64 						isDefault:true
 65 					},
 66 					{
 67 						text:"Close",
 68 						handler:
 69 							{
 70 								fn: function(e, btnId) { this._buttonPressed(btnId); },
 71 								obj: "close",
 72 								scope: this
 73 							}
 74 					} ]
 75 			});
 76 		this.textBox = document.createElement("TEXTAREA");
 77 		this.textBox.setAttribute("rows", "20");
 78 		this.textBox.setAttribute("cols", "80");
 79 		this.textBox.setAttribute("wrap", "off");
 80 		this.textBox.setAttribute("id", divName + "__BodyTextArea");
 81 		this.textBox.setAttribute("name", divName + "__BodyTextArea");
 82 		this.textBox.setAttribute("class", "screenWide lightBox");
 83 		this.dialog.setBody(this.textBox);
 84 		this.dialog.render(document.body);
 85 		var oButtons = this.dialog.getButtons();
 86 		this._btnOk = oButtons[0];
 87 		this._btnCancel = oButtons[1];
 88 		this._oPendingCallback = null;
 89 	};
 90 	RXBuild.UI.Dialog.prototype.constructor = RXBuild.UI.Dialog;
 91 	/** Performs the appropriate action when a button is pressed
 92 		@private
 93 	*/
 94 	RXBuild.UI.Dialog.prototype._buttonPressed = function(buttonId) {
 95 		var oTemp = this._oPendingCallback;
 96 		this._oPendingCallback = null;
 97 		this.dialog.hide();
 98 		var oContent = this._oContentChild;
 99 		if (this._oContentChild) {
100 			this._oContentChild = null;
101 			this.dialog.setBody(this.textBox);
102 		}
103 		if (!oContent) oContent = this.textBox.value;
104 		if (buttonId == "ok" && oTemp) {
105 			oTemp(oContent);
106 		}
107 	};
108 	/** Displays modally the dialog box with the provided header, text and buttons.
109 		@param {String} headerHTML HTML content to display in the header of the dialog
110 		@param {String|HTMLElement} defaultText The main content to display in the dialogs body initially
111 		@param {String[]|String} verbs Optional. An array of labels to put on the OK and Cancel buttons (respectively)
112 		@param {Function} callback Optional. The method to call when the user validates the dialog.
113 		@param {object} callbackContext Optional. The object to scope before calling callback
114 	*/
115 	RXBuild.UI.Dialog.prototype.show = function(headerHTML, defaultText, verbs, callback, callbackContext) {
116 		if (this._oPendingCallback != null) throw "There is already a dialog expecting a response being shown.";
117 		this.dialog.header.innerHTML = headerHTML;
118 		if (typeof(defaultText) == "string") {
119 			this.dialog.setBody(this.textBox);
120 			this.textBox.value = defaultText;
121 		} else {
122 			this._oContentChild = defaultText;
123 			this.dialog.setBody(this._oContentChild);
124 		}
125 		if (callback && callbackContext)
126 			callback = RXBuild.Utils.createDelegate(callbackContext, callback);
127 		this._oPendingCallback = callback;
128 		if (typeof(defaultText) == "string") {
129 			if (callback) {
130 				this.textBox.removeAttribute("readonly");
131 				YAHOO.util.Event.removeListener(this.textBox, "click");
132 			} else {
133 				this.textBox.setAttribute("readonly", "true");
134 				YAHOO.util.Event.addListener(this.textBox, "click", RXBuild.Utils.createDelegate(this, function() {
135 					if (this.textBox.select) this.textBox.select();
136 				}));
137 			}
138 		}
139 		var sV1 = "Ok";
140 		var sV2 = callback ? "Cancel" : "Close";
141 		if (verbs && verbs.length == 0)
142 			verbs = null;
143 		if (verbs) {
144 			if (typeof(verbs) == "object" && verbs.length == 1)
145 				verbs = verbs[0];
146 			if (typeof(verbs) == "string")
147 				if (callback)
148 					sV1 = verbs;
149 				else
150 					sV2 = verbs;
151 			else {
152 				sV1 = verbs[0];
153 				sV2 = verbs[1];
154 			}
155 		}
156 		this._btnOk.set("label", sV1);
157 		this._btnCancel.set("label", sV2);
158 		this._btnOk.setStyle("display", callback ? "" : "none");
159 		this.dialog.show();
160 		this.textBox.focus();
161 		if (!callback)
162 			this.textBox.focus();
163 	};
164 
165 })();
