 31   var getPageTopLeft = function(el) {
 32 	  var res = {x: 0, y: 0};
 33 	  while (el) {
 34 	    res.x += el.offsetLeft;
 35 	    res.y += el.offsetTop;
 36 	    var parent = el.offsetParent;
 37 	    el = (el != parent) ? parent : null;
 38 	  }
 39 	  return res;
 40 	};

 43 		Creates a new instance of RXBuild.UI.Status
 44 		@class The RXBuild.UI.Status controls an always-on-top div that shows a tasks status
 45 		@constructor
 46 		@param {String} [canvasName] The name of the canvas to build

 48 		@param {HTMLElement} [parent=document.body] The container for the canvas

 50 	RXBuild.UI.Diagram = function (canvasName, className, parent) {
 51 		this.canvas = document.createElement("CANVAS");
 52 		if (className) this.canvas.setAttribute("class", className);
 53 		if (canvasName)	this.canvas.setAttribute("id", canvasName);
 54 		this.canvas.setAttribute("style", "top: 0px; left: 0px; padding: 0; margin: 0; border: 0;");
 55 		(parent ? parent : document.body).appendChild(this.canvas);
 56 		if (window.G_vmlCanvasManager) {
 57 			window.G_vmlCanvasManager.initElement(this.canvas);
 58 		}
 59 		this.context = this.canvas.getContext("2d");
 60 	};
 61 	RXBuild.UI.Diagram.prototype.constructor = RXBuild.UI.Diagram;
 62 	RXBuild.UI.Diagram.prototype.updateSize = function (size) {
 63 		if (size.w != this.canvas.width || size.h != this.canvas.height) {
 64 			this.canvas.style.width = size.w + "px";
 65 			this.canvas.style.height = size.h + "px";
 66 			this.canvas.width = size.w;
 67 			this.canvas.height = size.h;
 68 		} else
 69   		this.canvas.width = this.canvas.width;
 70 	};
 71 	RXBuild.UI.Diagram.prototype.clear = function () {
 72 	  if (this.renderable) this.renderable.remove();
 73 	  this.renderable = null;
 74 	  this.canvas.width = this.canvas.width;
 75 	};
 76 	RXBuild.UI.Diagram.prototype.render = function (renderable) {
 77 	  if (this.renderable) this.renderable.remove();
 78 	  this.renderable = renderable;
 79 	  var size = renderable.measure(this);
 80 	  this.updateSize(size);
 81 	  this.offset = getPageTopLeft(this.canvas);
 82 		renderable.draw(this, this.context, 0, 0, size);
 83 	};

128   /** Creates a new instance of RXBuild.UI.Diagram.Box (which inherits RXBuild.UI.Diagram.Renderable)
129     @class The RXBuild.UI.Diagram.Box draws the inner renderer with a box around it
130     @p
131     @base RXBuild.UI.Diagram.Renderable
132     @constructor
133     @param {} innerRenderable
134   */
135   RXBuild.UI.Diagram.Box = function (innerRenderable, padding, spacing) {
136     RXBuild.UI.Diagram.Renderable.call(this, innerRenderable);
137     this.debug = innerRenderable;
138     if (innerRenderable && !innerRenderable.measure) innerRenderable = new RXBuild.UI.Diagram.DomRenderer(innerRenderable);
139     this.innerRenderable = innerRenderable;
140     this.padding = padding || {w: 3, h: 3};
141     this.spacing = spacing || {w: 3, h: 3};
142   };
143   RXBuild.UI.Diagram.Box.prototype = new RXBuild.UI.Diagram.Renderable;
144   RXBuild.UI.Diagram.Box.prototype.constructor = RXBuild.UI.Diagram.Box;
145   var normalisePadding = function(padding) {
146     return {
147       x: (typeof padding.x == "undefined") ? padding.w : padding.x,
148       r: (typeof padding.r == "undefined") ? padding.w : padding.r,
149       y: (typeof padding.y == "undefined") ? padding.h : padding.y,
150       b: (typeof padding.b == "undefined") ? padding.h : padding.b
151     };
152   };
153   var inflateRect = function(orig, pad) {
154     var result = {
155       w: orig.w + pad.x + pad.r,
156       h: orig.h + pad.y + pad.b
157     };
158     if (typeof orig.x == "number") {
159       result.x = orig.x - pad.x;
160     }
161     if (typeof orig.y == "number") {
162       result.y = orig.y - pad.y;
163     }
164     return result;
165   };
166   var offsetRect = function(rect, pad) {
167     rect.x += pad.x;
168     rect.y += pad.y;
169   };
170   RXBuild.UI.Diagram.Box.prototype.measure = function(render) {
171     var child = this.innerRenderableSize = this.innerRenderable.measure(render);
172     this.size = inflateRect(inflateRect({w: child.w, h: child.h}, normalisePadding(this.padding)), normalisePadding(this.spacing));
173 		if (this.innerRenderable.baseOffset) this.baseOffset = this.innerRenderable.baseOffset;
174     return this.size;
175 	};
176   RXBuild.UI.Diagram.Box.prototype.remove = function() {
177     this.innerRenderable.remove();
178   };
179   RXBuild.UI.Diagram.Box.prototype.drawFrame = function (render, child, frame, overall) {
180     render.context.strokeRect(frame.x, frame.y, frame.w, frame.h);
181   };
182 	RXBuild.UI.Diagram.Box.prototype.draw = function(render, ctx, x, y, size) {
183 	  var childRect = {
184 	    x: x, y: y,
185 	    w: this.innerRenderableSize.w, h: this.innerRenderableSize.h
186 	  };
187 	  var spacing = normalisePadding(this.spacing);
188 	  var padding = normalisePadding(this.padding);
189 	  offsetRect(childRect, spacing);
190 	  offsetRect(childRect, padding);
191 	  var frameRect = inflateRect(childRect, spacing);
192 	  var totalRect = inflateRect(frameRect, padding);
193 	  this.drawFrame(render, childRect, frameRect, totalRect);
194     this.innerRenderable.draw(render, ctx, childRect.x, childRect.y, childRect);
195 	};

204   RXBuild.UI.Diagram.Align = function (children, vertical, spacing) {
205     this.children = children;
206     this.vertical = vertical;
207     this.spacing = typeof spacing == "undefined" ? 3 : spacing;
208   };
209   RXBuild.UI.Diagram.Align.prototype = new RXBuild.UI.Diagram.Renderable;
210   RXBuild.UI.Diagram.Align.prototype.constructor = RXBuild.UI.Diagram.Boxer;
211   RXBuild.UI.Diagram.Align.prototype.measure = function(render) {
212     var childSizes = this.childSizes = new Array(this.children.count);
213     var max = 0;
214     var running = 0;
215     for (var i = this.children.length - 1; i >= 0; i--){
216       var child = this.children[i];
217       var size = child.measure(render);
218       childSizes[i] = size;
219       if (this.vertical) {
220         if (size.w > max) max = size.w;
221         running += size.h;
222       } else {
223         var height = child.baseOffset || 0;
224         height = size.h + (height < 0 ? - height : height);
225         if (height > max) max = height;
226         running += size.w;
227       }
228     }
229     running += (this.children.length - 1) * this.spacing;
230     this.size = this.vertical ? {w: max, h: running} : {w: running, h:max};
231 		return this.size;
232 	};
233   RXBuild.UI.Diagram.Align.prototype.remove = function() {
234     for (var i = this.children.length - 1; i >= 0; i--){
235       this.children[i].remove();
236     }
237   };
238 	RXBuild.UI.Diagram.Align.prototype.draw = function(render, ctx, x, y, size) {
239 	  var v = this.vertical;
240 	  var running = 0;
241 	  var max = v ? this.size.w : this.size.h;
242     //console.dir({pos: [x, y], size: [this.size.w, this.size.h]});
243     //console.group(this.debug || (this.vertical ? "vertical" : "horizontal") + " align");
244 	  for (var i=0, iNumberOfChildren = this.children.length; i < iNumberOfChildren; i++) {
245 	    var child = this.children[i];
246 	    var childsize = this.childSizes[i];
247 	    var pos = { x: x, y: y };
248 	    if (v) {
249 	      pos.x += (max - childsize.w) / 2;
250 	      pos.y += running;
251 	    } else {
252 	      pos.x += running;
253 	      pos.y += (max - childsize.h) / 2 - (child.baseOffset || 0);
254 	    }
255       running += (v ? childsize.h : childsize.w) + this.spacing;
256 	    child.draw(render, ctx, pos.x, pos.y, size);
257 	  }  
258 	  //console.groupEnd();
259 	};
