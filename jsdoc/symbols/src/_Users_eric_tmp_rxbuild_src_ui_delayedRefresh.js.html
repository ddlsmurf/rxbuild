 18 /** @fileOverview Holds the async delayed cancellable class used for refreshes
 19 	@requires utils.js
 20 */
 21 if (!RXBuild)
 22 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 23 	var RXBuild = { };
 24 if (!RXBuild.UI)
 25 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 26 	 RXBuild.UI = {};
 27 
 28 (function() {
 29 
 30 	/** 
 31 		Creates a new instance of RXBuild.UI.DelayedRefresh
 32 		@class The RXBuild.UI.DelayedRefresh Holds a triggerable and cancelable delayed event object which adjusts its frequency.
 33 		@property {Number} defaultDelay The delay with which to raise the delegate if none is specified
 34 		@constructor
 35 		@param {Number} defaultDelay The default delay to use when none is specified.
 36 		@param {Function} callBack The main method to call after time has elapsed
 37 		@param {Function} invalidateCallback Optional. A function to call when a refresh is pending.
 38 		@param {Float} adaptiveSpeed A number from 0 to 1.0 that determines the hi-pass threshold for automatically updating
 39 						the defaultDelay. If set to 0, the defaultDelay never changes, if set to 1.0, it is always the duration
 40 						of the last execution of the callback, in between, the delay is averaged at each execution. A good value is 0.1.
 41 	*/
 42 	RXBuild.UI.DelayedRefresh = function (defaultDelay, callBack, invalidateCallback, adaptiveSpeed) {
 43 		this.defaultDelay = defaultDelay;
 44 		if (!adaptiveSpeed) adaptiveSpeed = 0;
 45 		if (adaptiveSpeed < 0) adaptiveSpeed = 0;
 46 		if (adaptiveSpeed > 1.0) adaptiveSpeed = 1.0;
 47 		this.adaptiveSpeed = adaptiveSpeed;
 48 		this.nextTimeout = 0;
 49 		this.pendingEvent = null;
 50 		this.callBack = callBack;
 51 		this.invalidateCallback = invalidateCallback;
 52 	};
 53 	RXBuild.UI.DelayedRefresh.prototype.constructor = RXBuild.UI.DelayedRefresh;
 54 	/** Cancel the pending trigger
 55 		@return {Boolean} True if there was an event pending to cancel
 56 	*/
 57 	RXBuild.UI.DelayedRefresh.prototype.stop = function() {
 58 		this.nextTimeout = 0;
 59 		if (this.pendingEvent)
 60 			{
 61 				window.clearTimeout(this.pendingEvent);
 62 				this.pendingEvent = null;
 63 				return true;
 64 			}
 65 		return false;
 66 	};
 67 	/** Calls the delegate and updates the timer frequency
 68 		@private
 69 	*/
 70 	RXBuild.UI.DelayedRefresh.prototype.__runDelegate = function() {
 71 		this.pendingEvent = null;
 72 		var iStart = new Date();
 73 		try {
 74 			this.callBack();
 75 		} catch (e) {
 76 			window.alert("Warning - CallBack raised an exception\n" + e);
 77 		}
 78 		var iDuration = ((new Date()).getTime() - iStart) * 10;
 79 		if (this.adaptiveSpeed != 0){
 80 			this.defaultDelay = (1.0 - this.adaptiveSpeed) * this.defaultDelay + this.adaptiveSpeed * iDuration;
 81 			if (this.defaultDelay < 10) this.defaultDelay = 10;
 82 		}
 83 	};
 84 	/** Restarts the pending event, cancelling any pending events
 85 		@param {Number} timeout The timeout can override the internally maintained timeout
 86 	*/
 87 	RXBuild.UI.DelayedRefresh.prototype.reset = function(timeout) {
 88 		if (typeof timeout == "undefined")
 89 			timeout = this.defaultDelay;
 90 		var iNextTimeout = new Date((new Date()).getTime() + timeout);
 91 		var bShouldInvalidate = !this.pendingEvent;
 92 		this.stop();
 93 		if (bShouldInvalidate && this.invalidateCallback) this.invalidateCallback();
 94 		this.nextTimeout = iNextTimeout;
 95 		this.pendingEvent = window.setTimeout(RXBuild.Utils.createDelegate(this,
 96 			function() {
 97 				if (this.nextTimeout > iNextTimeout)
 98 					return;
 99 				this.__runDelegate();
100 			}), timeout);
101 	};
102 
103 
104 })();
