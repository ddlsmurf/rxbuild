 19 /**	@fileOverview Contains the component that displays an tree view of the regexps DOM
 20 	@requires utils.js
 21 	@requires regex_controller.js
 22 	@requires regex_engine.js
 23 */
 24 
 25 if (!RXBuild)
 26 	/** @namespace The RXBuild namespace is the root namespace for all things RXBuild */
 27 	var RXBuild = { };
 28 if (!RXBuild.UI)
 29 	/** @namespace The RXBuild.UI namespace is the root namespace for all things related to RXBuilds user interface */
 30 	 RXBuild.UI = {};
 31 
 32 (function() {
 33 	/** 
 34 		Creates a new instance of RXBuild.UI.HtmlExplanationTreeBuilder (which inherits RXBuild.Dom.NodeVisitor)
 35 		@class The RXBuild.UI.HtmlExplanationTreeBuilder visits the DOM of the parsed regexp to build a corresponding YUI tree
 36 		@p
 37 		@base RXBuild.Dom.NodeVisitor
 38 		@constructor
 39 	*/
 40 	RXBuild.UI.HtmlExplanationTreeBuilder = function (parentNode) {
 41 		RXBuild.Dom.NodeVisitor.call(this);
 42 		this.parentStack = [];
 43 		this.currentParent = parentNode;
 44 	};
 45 	RXBuild.UI.HtmlExplanationTreeBuilder.prototype = new RXBuild.Dom.NodeVisitor;
 46 	RXBuild.UI.HtmlExplanationTreeBuilder.prototype.constructor = RXBuild.UI.HtmlExplanationTreeBuilder;
 47 	/** Called when visiting a leaf node
 48 		@param {RXBuild.Dom.Node} leafNode The node being visited
 49 		@return {Boolean} Return true to continue visit
 50 	*/
 51 	RXBuild.UI.HtmlExplanationTreeBuilder.prototype.visit = function(leafNode) {
 52 		(new YAHOO.widget.TextNode(leafNode.GetHtml(true), this.currentParent, false)).data = leafNode;
 53 		return true;
 54 	};
 55 	/** Called when visiting a parent node
 56 		@param {RXBuild.Dom.Node} compositeNode The node being visited
 57 		@return {Boolean} Return true to visit children on this node
 58 	*/
 59 	RXBuild.UI.HtmlExplanationTreeBuilder.prototype.visitEnter = function(compositeNode) {
 60 		var oNewParent = (new YAHOO.widget.TextNode(compositeNode.GetHtml(true), this.currentParent, true));
 61 		oNewParent.data = compositeNode;
 62 		this.parentStack.push(this.currentParent);
 63 		this.currentParent = oNewParent;
 64 		return true;
 65 	};
 66 	/** Called when finished visiting a parent node
 67 		@param {RXBuild.Dom.Node} compositeNode The node that completed visit
 68 		@return {Boolean} Return true to continue the visit after the compositeNode
 69 	*/
 70 	RXBuild.UI.HtmlExplanationTreeBuilder.prototype.visitLeave = function(compositeNode) {
 71 		this.currentParent = this.parentStack.pop();
 72 		return true;
 73 	};
 74 	
 75 	/** 
 76 		Creates a new instance of RXBuild.UI.HtmlResultTreeViewer (which inherits RXBuild.Engine.ResultListener)
 77 		@class The RXBuild.UI.HtmlResultTreeViewer displays the results of matching a regular expression inline with the text
 78 		@property {HTMLElement} container The DOM element that holds the results for display		
 79 		@base RXBuild.Engine.ResultListener
 80 		@constructor
 81 		@param {} container
 82 	*/
 83 	RXBuild.UI.HtmlExplanationTreeViewer = function (container, id) {
 84 		RXBuild.Engine.ResultListener.call(this);
 85 		this.container = document.createElement("DIV");
 86 		this.container.className = "rxb-domtree";
 87 		this.container.setAttribute("id", id);
 88 		container.appendChild(this.container);
 89 		this.errorcontainer = document.createElement("DIV");
 90 		this.errorcontainer.setAttribute("id", id + "_errorContainer");
 91 		container.appendChild(this.errorcontainer);
 92 		this.tree = new YAHOO.widget.TreeView(this.container);
 93 	};
 94 	RXBuild.UI.HtmlExplanationTreeViewer.prototype.constructor = RXBuild.UI.HtmlExplanationTreeViewer;
 95 	/** Updates the view to reflect the provided regular expression and input
 96 		@param {RXBuild.RegExp} regex The new pattern
 97 		@param {String} input The sample input text in which to run the matches
 98 	*/
 99 	RXBuild.UI.HtmlExplanationTreeViewer.prototype.updateRegExp = function (regex, input) {
100 		this.tree.removeChildren(this.tree.getRoot());
101 		var oCompiledRegExp;
102 		try {
103 			oCompiledRegExp = regex.compile();
104 			this.container.style.display = "";
105 			this.errorcontainer.style.display = "none";
106 			var oVisitor = new RXBuild.UI.HtmlExplanationTreeBuilder(this.tree.getRoot());
107 			oCompiledRegExp.StartVisit(oVisitor);
108 			this.tree.draw();
109 		} catch (e) {
110 			this.container.style.display = "none";
111 			this.errorcontainer.style.display = "";
112 			this.errorcontainer.innerHTML = e.plainTextToHtml();
113 		}
114 	};
115 })();
