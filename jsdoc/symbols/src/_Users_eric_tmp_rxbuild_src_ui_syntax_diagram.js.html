 25 	var runInPath = function(ctx, stroke, fun, funContext) {
 26 	  ctx.beginPath();
 27     var strokeBackup = ctx.strokeStyle;
 28     if (stroke)
 29       ctx.strokeStyle = stroke;
 30     var res = fun.call(funContext || this, ctx);
 31     ctx.stroke();
 32     ctx.closePath();
 33     if (stroke)
 34   		ctx.strokeStyle = strokeBackup;
 35   	return res;
 36 	};

 49 	var drawArrowCap = function(ctx, pos, dir) {
 50 	  runInPath(ctx, null, function(ctx) {
 51   		var pPoint1 = {
 52   			x: pos.x + [-arrowSize, halfArrow, arrowSize, -halfArrow][dir],
 53   			y: pos.y + [-halfArrow, -arrowSize, halfArrow, arrowSize][dir]
 54   		};
 55   		var pPoint2 = {
 56   			x: pos.x + [-arrowSize, -halfArrow, arrowSize, halfArrow][dir],
 57   			y: pos.y + [halfArrow, -arrowSize, -halfArrow, arrowSize][dir]
 58   		};
 59   		ctx.moveTo(pPoint1.x, pPoint1.y);
 60   		ctx.lineTo(pos.x, pos.y);
 61   		ctx.lineTo(pPoint2.x, pPoint2.y);
 62   	});
 63 	};

 80 	  arrow: function(context, start, stop, color, withcap) {
 81   		var endDirection = 
 82   				 (start.y < stop.y ? dir.down :
 83   				 (start.x < stop.x ? dir.right :
 84   				 (start.x > stop.x ? dir.left :
 85   					dir.up))); // Yeah that's right, so dont try any fancy-schmancy angles, you'll be disappointed
 86   	  runInPath(context, color, function(ctx) {
 87     		ctx.moveTo(start.x, stop.y);
 88     		ctx.lineTo(stop.x, stop.y);
 89   	  });
 90   	  if (withcap || typeof withcap == "undefined")	drawArrowCap(context, stop, endDirection);
 91   	},
 92   	corneredLine: function(context, start, stop, startVertical, drawCap, color) {
 93   		var startDirection = startVertical ? (start.y < stop.y ? dir.down : dir.up) :
 94   							 (start.x < stop.x ? dir.right : dir.left);
 95   		var endDirection = (!startVertical) ? (start.y < stop.y ? dir.down : dir.up) :
 96   							 (start.x < stop.x ? dir.right : dir.left);
 97   	  runInPath(context, color, function(ctx) {
 98     		var firstCorner = {
 99     			x: startVertical ? start.x : stop.x + (startDirection == dir.left ? lineAngleRadius : -lineAngleRadius),
100     			y: startVertical ? stop.y + (startDirection == dir.up ? lineAngleRadius : -lineAngleRadius) : start.y
101     		};
102     		var circleCenter = {
103     			x: startVertical ? (start.x + (endDirection == dir.left ? -lineAngleRadius : lineAngleRadius)) : firstCorner.x,
104     			y: startVertical ? firstCorner.y : (start.y + (endDirection == dir.up ? -lineAngleRadius : lineAngleRadius))
105     		};
106     		var startAngle = ([2, 3, 0, 1][endDirection]  * Math.PI) / 2.0;
107     		var endAngle = (startDirection  * Math.PI) / 2.0;
108     		var isAntiClockWise =
109     			(startDirection == dir.up && endDirection == dir.left || startDirection == dir.down && endDirection == dir.right ||
110     			 startDirection == dir.left && endDirection == dir.down || startDirection == dir.right && endDirection == dir.up);
111     		ctx.moveTo(start.x, start.y);
112     		ctx.lineTo(firstCorner.x, firstCorner.y);
113     		ctx.arc(circleCenter.x, circleCenter.y, lineAngleRadius, startAngle, endAngle, isAntiClockWise);
114     		ctx.lineTo(stop.x, stop.y);
115   		});  
116   		if (drawCap) drawArrowCap(context, stop, endDirection);
117   	},
118   	connect: function(render, a, b, alignLeft) {
119   	  if (a.connectors) a = a.connectors[1];
120   	  if (b.connectors) b = b.connectors[0];
121 
122   	  if (typeof a.length == "number") {
123   	    for (var i = a.length - 1; i >= 0; i--)
124   	      this.connect(render, a[i], b, alignLeft);
125   	    return;
126   	  }
127   	  if (typeof b.length == "number") {
128   	    for (var i = b.length - 1; i >= 0; i--)
129   	      this.connect(render, a, b[i], alignLeft);
130   	    return;
131   	  }
132 
133   	  var dir = b.y - a.y;
134   	  var color = null;
135 
136   	  if (dir == 0) render.plot.arrow(render.context, a, b, color, false);
137   	  else {
138   	    if (b.x < a.x) {
139   	      var c = b;
140   	      b = a;
141   	      a = c;
142   	    }
143   	    var mid = { x: alignLeft ? a.x + lineAngleRadius : b.x - lineAngleRadius, y: a.y + (b.y - a.y) / 2};
144   	    this.corneredLine(render.context, a, mid);
145   	    this.corneredLine(render.context, mid, b, true);
146   	  }
147   	}

188     RXBuild.UI.Diagram.Renderable.call(this, box);
189     this.debug = box;

191     if (label && !label.measure) label = new RXBuild.UI.Diagram.DomRenderer(label);

198   RXBuild.UI.SyntaxDiagram.Repeated.prototype.constructor = RXBuild.UI.SyntaxDiagram.Repeated;
199   RXBuild.UI.SyntaxDiagram.Repeated.prototype.measure = function(render) {
200     var boxSize = this.boxSize = this.box.measure(render);
201     var labelSize = this.labelSize = this.label ? this.label.measure(render) : {w: 0, h: 0};
202 	  var lineHeight = labelSize.h + 2 * render.plot.config.lineAngleRadius;
203 	  var labelWidth = this.label ? labelSize.w + render.plot.config.lineAngleRadius * 2 : 0;
204 	  this.baseOffset = this.direction == dir.up ? lineHeight / 2 : -lineHeight / 2;
205 	  this.baseOffset += this.box.baseOffset || 0;
206 		this.size = {
207 		  w: (labelWidth > boxSize.w ? labelWidth : boxSize.w) + render.plot.config.arrowSize * 2,
208 		  h: boxSize.h + lineHeight
209 		};
210     return this.size;
211 	};
212   RXBuild.UI.SyntaxDiagram.Repeated.prototype.remove = function() {
213     this.box.remove();
214     if (this.label) this.label.remove();
215   };
216 	RXBuild.UI.SyntaxDiagram.Repeated.prototype.draw = function(render, ctx, x, y, size) {
217 	  var lineHeight = this.labelSize.h + 2 * render.plot.config.lineAngleRadius;
218 	  
219     this.box.draw(render, ctx,
220       x + (this.size.w - this.boxSize.w) / 2,
221       y + (this.direction == dir.up ? lineHeight : 0), size);
222 
223     var line = lineHeight / 2 + (this.direction == dir.up ? 0 : this.boxSize.h);
224     var center = { x: x + this.size.w / 2, y: y + this.size.h };
225     var mid1 = { x: center.x + this.labelSize.w / 2, y: y + line};
226     var mid2 = { x: center.x - this.labelSize.w / 2, y: y + line};
227     
228     var cx = this.connectors = [
229       {x: x, y: this.box.connectors[0].y},
230       {x: x + this.size.w, y: this.box.connectors[1].y}];
231     
232     render.plot.corneredLine(render.context, mid1, {x: cx[1].x - render.plot.config.arrowSize / 2, y: cx[1].y}, false, this.leftToRight);
233     render.plot.corneredLine(render.context, mid2, {x: cx[0].x + render.plot.config.arrowSize / 2, y: cx[0].y}, false, !this.leftToRight);
234     
235     var color = null;
236     render.plot.runInPath(ctx, color, function(ctx) {
237   		ctx.moveTo(x, cx[0].y);
238   		ctx.lineTo(this.box.connectors[0].x, cx[0].y);
239   		ctx.moveTo(this.box.connectors[1].x, cx[1].y);
240   		ctx.lineTo(cx[1].x, cx[1].y);
241 	  }, this);
242     
243     if (this.label) {
244       this.label.draw(render, ctx,
245         x + (this.size.w - this.labelSize.w) / 2,
246         y + line - this.labelSize.h / 2, size);
247     }
248     //  console.dir({id: this.debug, pos: [x, y], size: [this.size.w, this.size.h]});
249 	};

309 	/** Updates the view to reflect the provided regular expression and input
310 		@param {RXBuild.RegExp} regex The new pattern
311 		@param {String} input The sample input text in which to run the matches
312 	*/
313 	RXBuild.UI.DiagramViewer.prototype.updateRegExp = function (regex, input) {
314 	  this.canvas.clear();
315 		var oCompiledRegExp;
316 		try {
317 			oCompiledRegExp = regex.compile();
318 			this.container.style.display = "";
319 			this.errorcontainer.style.display = "none";
320 			this.canvas.render(oCompiledRegExp.buildGrammarDiagram(this.container));
321 		} catch (e) {
322 			this.container.style.display = "none";
323 			this.errorcontainer.style.display = "";
324 			this.errorcontainer.innerHTML = e.toString().plainTextToHtml();
325 		}
326 	};
