 31 /** @class Abstract base class for all matching nodes in the regexps DOM tree
 32 	@property {RXBuild.Dom.Node} next The node to match following the current instance
 33 	@property {Number} id A unique (in the regexp) numeric ID for this item
 34 	@property {regex_parser.TokenInfo[]} tokens An array of tokens from the input stream that created this instance
 35 */
 36 RXBuild.Dom.Node = function ()
 37 {
 38 	this.next = null;
 39 	this.src = null;
 40 	this.id = null;
 41 	this.tokens = new Array();
 42 };
 43 
 44 /** @private
 45 Executes this match	
 46 */
 47 RXBuild.Dom.Node.prototype.Match = function() {return false;};
 48 
 49 /** Returns a human readable description of this match
 50 	@return {String} The description of this node in human readable format.
 51 */
 52 RXBuild.Dom.Node.prototype.GetDescription = function() {return "-RXBuild.Dom.Node-";};
 53 /** Adds all tokens passed in all arguments to the tokens property of this instance
 54 	@param {regex_parser.TokenInfo} ... Tokens to include in this instance
 55 */
 56 RXBuild.Dom.Node.prototype.AddTokens = function() {
 57 	for (var i = 0; i < arguments.length; i++)
 58 		{
 59 			var obj = arguments[i];
 60 			if (obj instanceof Array)
 61 				for (var j=0; j < obj.length; j++)
 62 					this.AddTokens(obj[j]);
 63 			else
 64 				this.tokens.push(obj);
 65 		}
 66 };
 67 
 68 /** Returns a human readable description of this match and its following matches
 69 	@return {String} The description of this node and the following in human readable format.
 70 */
 71 RXBuild.Dom.Node.prototype.GetChainDescription = function() {
 72 	var sResult = "";
 73 	var oItem = this;
 74 	while (oItem != null)
 75 	{
 76 		if (sResult != "") sResult += ", ";
 77 		sResult += oItem.GetDescription();
 78 		oItem = oItem.next;
 79 	}
 80 	return sResult;
 81 };
 82 /** Finds and returns the last item in the chain starting from the current instance
 83 	@return {RXBuild.Dom.Node} The last node in this nodes siblings
 84 */
 85 RXBuild.Dom.Node.prototype.GetLast = function() {
 86 	var oItem = this;
 87 	while (oItem.next != null)
 88 		oItem = oItem.next;
 89 	return oItem;
 90 };
 91 /** Helper method allowing the DOM to transform itself
 92 	@private
 93 	@return {RXBuild.Dom.Node} A reference to the node that should replace this instance
 94 */
 95 RXBuild.Dom.Node.prototype.Flatten = function() {
 96 	if (this.next) this.next = this.next.Flatten();
 97 	return this;
 98 };
 99 /** Adds the provided RXBuild.Dom.Node to the end of this items sibling chain
100 	@param {RXBuild.Dom.Node} newLast The node to append.
101 */
102 RXBuild.Dom.Node.prototype.AppendAtEnd = function(newLast) {
103 	var oItem = this.GetLast();
104 	oItem.next = newLast;
105 };
106 /** Copies base information from the provided node into the current instance
107 	@private
108 */
109 RXBuild.Dom.Node.prototype.CopyFrom = function(prevNode) {
110 	this.next = prevNode.next; 
111 	this.id = prevNode.id;
112 	this.tokens = prevNode.tokens;
113 };

115 /** Returns an HTML description of this match (Must be overriden - this returns a fat error)
116 	@return {String} An HTML description of this match
117 */
118 RXBuild.Dom.Node.prototype.GetHtml = function() {
119 	return "<h1>Error - RXBuild.Dom.Node.GetHtml sifted through on " + RXBuild.Utils.getObjectDesc(this) + "</h1>";

122 /** Used to display this nodes HTML in an interactive way to the user - to be rewritten */
123 RXBuild.Dom.Node.prototype.GetTokenHighlightJS = function() {
124 	var sResult = "";
125 	for (var i = 0; i < this.tokens.length; i++)
126 		sResult += (i>0 ? ",[" : "[") + this.tokens[i].offset + "," + this.tokens[i].value.length + "]";
127 	return sResult;
128 };
129 /** Used to display this nodes HTML in an interactive way to the user - to be rewritten */
130 RXBuild.Dom.Node.NodeRegExTokenHighlight = function(tokenlist) {
131 	function IsCharInTokenList(c, tokenlist) {
132 		if (tokenlist == null) return true;
133 		for (var i=0; i < tokenlist.length; i++) {
134 			if (c >= tokenlist[i][0] && c < (tokenlist[i][0] + tokenlist[i][1]))
135 				return true;
136 		}
137 		return false;
138 	}
139 	var i = 0;
140 	var oObj;
141 	while ((oObj = document.getElementById("rxchr_" + i++)) != null) {
142 		oObj.style.color = IsCharInTokenList(i - 1, tokenlist) ? "black" : "grey";
143 	}
144 	window.event.stopPropagation();
145 }
146 /** Used to display this nodes HTML in an interactive way to the user - to be rewritten */
147 RXBuild.Dom.Node.prototype.GetHtmlOpenTag = function() {
148 	return "<li onmouseover=\"RXBuild.Dom.Node.NodeRegExTokenHighlight([" + this.GetTokenHighlightJS() + "]);\" onmouseout=\"RXBuild.Dom.Node.NodeRegExTokenHighlight(null);\">" + (this.id != null ? "<span class=\"rx_id\">" + this.id + ")</span> " : "");
149 };
150 /** Used to display this nodes HTML in an interactive way to the user - to be rewritten */
151 RXBuild.Dom.Node.prototype.GetChainHtml = function() {
152 	var sResult = "<ul>";
153 	var oItem = this;
154 	while (oItem != null)
155 	{
156 		sResult += oItem.GetHtmlOpenTag() + oItem.GetHtml() + "</li>";
157 		oItem = oItem.next;
158 	}
159 	return sResult + "</ul>";
160 };
161 /** Visits this node with the provided visitor
162 	@param {Function} func The function to call with each node as first parameter
163 	@param {object} param The second parameter to func
164 	@return {object} Whatever func last returned
165 */
166 RXBuild.Dom.Node.prototype.RunOnMe = function(func,param) {
167 	return func.call(this,param);
168 };
169 /** Calls RunOnMe on each node in this instances chain
170 @param {Function} func The function to call with each node as first parameter
171 @param {object} param The second parameter to func
172 @return {object} Whatever func returned for this instance
173 */
174 RXBuild.Dom.Node.prototype.RunForAll = function(func,param) {
175 	var oResult = this.RunOnMe(func,param);
176 	if (this.next != null)
177 		this.next = this.next.RunForAll(func,param);
178 	return oResult;
179 };
