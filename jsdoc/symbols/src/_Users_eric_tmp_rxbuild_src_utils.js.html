 21 /**
 22 	Returns a version of the string that is suitable for insertion in XML code
 23 	(& is replaced by &amp; and so on)
 24 	@type String
 25 */
 26 String.prototype.escapeXML = function () {
 27         return(
 28             this.replace(/&/g,'&')
 29                 .replace(/>/g,'>')
 30                 .replace(/</g,'<')
 31                 .replace(/'/g,''')
 32                 .replace(/"/g,'"')
 33         );
 34     };
 35 /**
 36 	Returns a version of the string that is has all standard 5 entities replaced
 37 	(&amp; is replaced by & and so on)
 38 	@type String
 39 */
 40 String.prototype.unescapeXML = function () {
 41         return(
 42             this
 43                 .replace(/>/g,'>')
 44                 .replace(/</g,'<')
 45                 .replace(/'/g,'\'')
 46                 .replace(/"/g,'\"')
 47 				.replace(/&/g,'&')
 48         );
 49     };

 63 /**
 64 	Returns a version of the string that is suitable for insertion in C/JS type code (without quotes though)
 65 	@type String
 66 
 67 */
 68 String.prototype.escapeToBackslashes = function () {
 69        return(
 70            (this.replace(/\\/g,'\\\\')).
 71                replace(/'/g,'\\\'').
 72                replace(/"/g,'\\\"').
 73                replace(/\t/g,'\\t').
 74                replace(/\r/g,'\\r').
 75                replace(/\f/g,'\\f').
 76                replace(/\v/g,'\\v').
 77                replace(/\n/g,'\\n').
 78 			   replace(/[^a-zA-Z0-9_+*\/\\'" .,?!@#$%^&*():;\[\]{}|><~-]/gi, function (s) {
 79 				var sHexCode = (new Number(s.charCodeAt(0))).toString(16);
 80 				if (sHexCode.length == 1) sHexCode = "0" + sHexCode;
 81 				if (sHexCode.length == 2) return "\\x" + sHexCode;
 82 				if (sHexCode.length == 3) sHexCode = "0" + sHexCode;
 83 				if (sHexCode.length == 4) return "\\u" + sHexCode;
 84 				return "[Unexplainable char 0x" + sHexCode + "]"; })
 85        );
 86    };

 97 /**
 98 	Returns the number of identical characters in this string and the one in s
 99 	starting from the beginning.
100 	@param {String} s The other string to compare to
101 	@param {Number} startIndex Optional - The position from which to start
102 	@return {Number} The number of common characters between the two strings
103 */
104 String.prototype.findCommonPrefix = function (s, startIndex) {
105 	var i;
106 	if (typeof(startIndex) != "number") startIndex = 0;
107 	for (i = startIndex; i < this.length && i < s.length && this.charAt(i) == s.charAt(i); i++)
108 		;
109   return i - startIndex;
110 };

114 	starting from the beginning using case insensitive .
115 	@param {String} s The other string to compare to
116 	@param {Number} startIndex Optional - The position from which to start
117 	@return {Number} The number of common characters between the two strings
118 */

120 	return this.toLocaleLowerCase().findCommonPrefix(s.toLocaleLowerCase());

178 /** Parses the current instance for url encoded key value pairs, in the form <code>key%20name=key%20value&key%20bool&key%20empty=</code>
179 	@return {Array of arrays of string} Returns an array with each element being a key and optional value pair.
180 */
181 String.prototype.parseURLKeyValues = function () {
182         var sItems = this.split(/&/);
183 		for (var i=0; i < sItems.length; i++) {
184 			var sPieces = sItems[i].split(/\=/);
185 			for (var j = sPieces.length - 1; j >= 0; j--){
186 				sPieces[j] = unescape(sPieces[j]);
187 			}
188 			sItems[i] = sPieces;
189 		}
190 		return sItems;
191     };

193 (function() {
194 	/** @private */
195 	function replaceValues(oldValues, newValues) {
196 		function findItemByName(name) {
197 			for (var j = oldValues.length - 1; j >= 0; j--){
198 				var oVal = oldValues[j];
199 				if (typeof(oVal) == 'string'){
200 					if (oVal == name) return j;
201 				} else if (oVal[0] == name) return j;
202 			}
203 			return -1;
204 		}
205 		function replaceItemByName(name, newValue) {
206 			var iItemToDel = findItemByName(name);
207 			if (iItemToDel > -1) oldValues[iItemToDel] = newValue;
208 			else oldValues.push(newValue);
209 		}
210 		function deleteItemByName(name) {
211 			var iItemToDel = findItemByName(name);
212 			if (iItemToDel > -1) oldValues.splice(iItemToDel, 1);
213 		}
214 		for (var i = newValues.length - 1; i >= 0; i--){
215 			var oItem = newValues[i];
216 			if (typeof(oItem) == "string")
217 				deleteItemByName(oItem);
218 			else
219 				replaceItemByName(oItem[0], oItem);
220 		}
221 	}
222 	/** @private */
223 	function encodeURLKeyValuePair(values) {
224 		var sResult = [];
225 		for (var i=0; i < values.length; i++) {
226 			var oItem = values[i];
227 			if (typeof(oItem) == "string")
228 				sResult.push(escape(oItem));
229 			else {
230 				var sItemString = [];
231 				for (var j=0; j < oItem.length; j++) {
232 					sItemString.push(escape(oItem[j]));
233 				}
234 				sResult.push(sItemString.join("="));
235 			}
236 		}
237 		return sResult.join("&");
238 	}
239 	
240 	/** Replaces the provide key value pairs, in the format as returned by parseURLKeyValues in the current instance
241 		and returns the resulting string.
242 		@param {Array of Array of string} newValues The values to edit in the current instance. The following formats are acceptable:
243 		<ul>
244 			<li><code>["abc"]</code> - to delete the key with name abc</li>
245 			<li><code>[["abc"]]</code> - to empty the key abc (but keep it present)</li>
246 			<li><code>[["abc", "def"]]</code> - to replace or add the key named abc with the value def</li>
247 		</ul>
248 		@return {String} A string encoded in the format <code>key%20name=key%20value&key%20bool&key%20empty=</code>
249 	*/
250 	String.prototype.replaceURLKeys = function(newValues) {
251 		var oOldValue = this.parseURLKeyValues();
252 		replaceValues(oOldValue, newValues);
253 		return encodeURLKeyValuePair(oOldValue);
254 	};

264 /** Tries really hard to build a flat array with all parameters
265 	If the first argument is an array, it will be the one to which the other items are added.
266 	You can call JoinArrays(null, ...) to make sure you get a new array. It returns null if the 
267 	result would have been empty.
268 	@param ... {Array} A list of arrays.
269 	@type Array
270 	@static
271 */
272 RXBuild.Utils.JoinArrays = function () {
273 	var oArg = Array.prototype.slice.call(arguments);
274 	var oResult = oArg[0];
275 	var i = 1;
276 	if (oResult != null && !(oResult instanceof Array))
277 	{
278 		i = 0;
279 		oResult = null;
280 	}
281 	if (oResult == null)
282 		oResult = [];
283 	for (; i < oArg.length; i++)
284 		{
285 			var oItem = oArg[i];
286 			if (oItem != null)
287 				if (oItem instanceof Array)
288 					for (var j=0; j < oItem.length; j++)
289 						RXBuild.Utils.JoinArrays(oResult, oItem[j]);
290 				else
291 					oResult.push(oItem);
292 		}
293 	if (oResult.length == 0)
294 		return null;
295 	return oResult;
296 };

308 /** Recreates and returns all the text in all nodes under the provided item
309 	@param {XMLNodeList|XMLNode|Array} item The root item whos content we want
310 	@return {String} All text content in the provided item
311 */
312 RXBuild.Utils.getXMLInnerText = function(item) {
313 	function addXMLInnerText(item, output) {
314 		if (!item) return;
315 		if (item.nodeValue) {
316 			output.push(item.nodeValue);
317 		} else if (item.length) {
318 			for (var i=0; i < item.length; i++) {
319 				addXMLInnerText(item[i], output);
320 			}
321 		} else if (item.hasChildNodes && item.hasChildNodes()) {
322 			addXMLInnerText(item.childNodes, output);
323 		}
324 	}
325 	var oResult = [];
326 	addXMLInnerText(item, oResult);
327 	return oResult.join("");

330 /**
331 	Generates a recursive print out of the provided object (o), skipping any properties
332 	contained in skip, and functions (unless requested).
333 	@param {object} o The object to explore.
334 	@param {String[]} skip A list of names of properties to exclude from the result
335 	@param {boolean} includeFunctions If set to true, function names are printed out too
336 	@param {String} i Initial indent string. (you can set to null)
337 	@type String
338 	@static
339 */
340 RXBuild.Utils.getObjectDesc = function(o, skip, includeFunctions, i) {
341 	function arrayHasString(arr, txt) {
342 		for (var s in arr) if (arr[s] == txt) return true;
343 		return false;
344 	}
345 	if (i == null) i = "";
346 	if (o == null) return "<null>";
347 	var oType = typeof(o);
348 	if (oType == "string") return "'" + o.escapeToBackslashes() + "'";
349 	if (oType == "number" || oType == "boolean") return o;
350 	if (oType == "function") return "{...}";
351 	var sResult = oType + "\n" + i;
352 	var bFirst = true;
353 	for (var prop in o)
354 		if (o.hasOwnProperty(prop) &&
355 			(includeFunctions || typeof(o[prop]) != "function") &&
356 			(skip == null || !arrayHasString(skip, prop))) {
357 			sResult += (bFirst ? "" : "\n" + i) + "-" + prop + ": " + RXBuild.Utils.getObjectDesc(o[prop], skip, includeFunctions, i + "  |");
358 			if (bFirst) bFirst = false;
359 		}
360 	return sResult;
361 	};

363 /** Builds an optimised regular expression to parse a list of tokens.
364 	@param {String[]} tokens An array of tokens to match
365 	@param {Boolean} noncapturing If set to true, generated groups are non-capturing (ie: <em>(?:)</em>)
366 	@param {Boolean} caseinsensitive If set to true, tokens are considered case insensitive
367 	@return {String} The regular expression that would match those tokens
368 */
369 RXBuild.Utils.buildRegexpFromTokens = function (tokens, noncapturing, caseinsensitive) {
370 	var root = [];
371 	function addNode(parent, str) { //Very dumb implementation of a suffix tree (but its home made - so dont you mention no fancy log(n) mumbo =)
372 		for (var i = 0; i < parent.length; i++) {
373 			var iCommon = caseinsensitive ? str.findCommonPrefixCaseInsensitive(parent[i].s) : str.findCommonPrefix(parent[i].s);
374 			if (iCommon == str.length) {
375 				// Found my terminal node
376 				parent[i].terminal = true;
377 				return ;
378 			} else if (iCommon > 0) {
379 				if (iCommon == parent[i].s.length) {
380 					// Found a node I should be under
381 					if (!parent[i].children)
382 						parent[i].children = [];
383 					addNode(parent[i].children, str.substr(iCommon));
384 					return;
385 				} else {
386 					// Found a node I must split and get under
387 					var sCommonPart = str.substr(0, iCommon);
388 					var oNodeToSplit = parent[i];
389 					var oNewChildFromPrevious = {
390 						s: oNodeToSplit.s.substr(iCommon)
391 						};
392 					if (oNodeToSplit.terminal) oNewChildFromPrevious.terminal = true;
393 					if (oNodeToSplit.children) oNewChildFromPrevious.children = oNodeToSplit.children;
394 					var sSuffix = str.substr(iCommon);
395 					var oNewChild = {
396 						s: sCommonPart,
397 						children: [oNewChildFromPrevious]
398 						};
399 					parent[i] = oNewChild;
400 					if (sSuffix.length == 0)
401 						oNewChild.terminal = true;
402 					else
403 						addNode(oNewChild.children, sSuffix);
404 					return;
405 				}
406 			}
407 		}
408 		// Couldnt find a place for myself, so made myself cosy
409 		parent.push({s: str, terminal: true});
410 	}
411 	function buildRegExp(parent, res, noncapturing) {
412 		for (var i=0; i < parent.length; i++) {
413 			if (i > 0) res.push("|");
414 			if (parent[i].children) {
415 				res.push(parent[i].s.escapeRegexp());
416 				res.push(noncapturing ? "(?:" : "(");
417 				buildRegExp(parent[i].children, res, noncapturing);
418 				res.push(")");
419 			} else {
420 				if (!parent[i].terminal) throw ("ASSERT: (This should never happen) Node is not terminal and has no children: '" + parent[i].s + "'");
421 				res.push(parent[i].s.escapeRegexp());
422 			}
423 			if (parent[i].children && parent[i].terminal) {
424 				res.push("?");
425 			}
426 		}
427 	}
428 	tokens = tokens.sort();
429 	for (var i = tokens.length - 1; i >= 0; i--){
430 		if (tokens[i].trim() == "")
431 			tokens.pop();
432 		else
433 			tokens[i] = tokens[i].trim();
434 	}
435 	for (var j=0; j < tokens.length; j++)
436 		addNode(root, tokens[j]);
437 	var oRes = [];
438 	buildRegExp(root, oRes, noncapturing);
439 	return "\\b" + oRes.join("") + "\\b";
440 };

442 /** Creates a new instance of RangeSet
443   @class The RangeSet holds a set of numeric ranges sparsely
444   @property {Boolean} inverted True if ranges contained in this instance are to be excluded
445   @property {Array} ranges A sorted array of objects in the form {f: num, t: num} specifying inclusive bounds
446   @constructor
447   @param {boolean} inverted <code>true</code> if ranges added to this instance are to be considered excluded, otherwise <code>false</code>.
448 */
449 RXBuild.Utils.RangeSet = function (inverted) {
450   this.ranges = [];
451   this.inverted = !!inverted;
452 };
453 RXBuild.Utils.RangeSet.prototype.constructor = RXBuild.Utils.RangeSet;
454 /** The add method adds the specified range to the current instance.
455   @param {Number|Range} range A range (of the form <code>{f: first_int, t: last_int}</code>), or a single included number.
456 */
457 RXBuild.Utils.RangeSet.prototype.add = function (range) {
458   function overlaps(a, b) {
459     return !(a.t < b.f || a.f > b.t);
460   }
461   function normalise(r) {
462     if (r.f > r.t) {
463       var f = r.f;
464       r.f = r.t;
465       r.t = f;
466     }
467     return r;
468   }
469   function mergeIfOverlaps(original, added) {
470     if (overlaps(original, added)) {
471       if (added.f < original.f) original.f = added.f;
472       if (added.t > original.t) original.t = added.t;
473       return true;
474     } else if (original.f == added.t + 1) {
475       original.f = added.f;
476       return true;
477     } else if (original.t == added.f - 1) {
478       original.t = added.t;
479       return true;
480     }
481     return false;
482   }
483   if (typeof range == "number") {
484     if (arguments.length == 1)
485       range = {f: range, t: range};
486     else range = normalise({f: range, t: arguments[1]});
487   }
488   else normalise(range);
489   var iNumberOfRanges = this.ranges.length;
490   for (var i=iNumberOfRanges; i > 0; i--) {
491     var oRange = this.ranges[iNumberOfRanges - i];
492     if (mergeIfOverlaps(oRange, range)) {
493       if (i > 1 && mergeIfOverlaps(oRange, this.ranges[iNumberOfRanges - i + 1])) {
494           this.ranges.splice(iNumberOfRanges - i + 1, 1);
495       }
496       return;
497     }
498     if (range.t < oRange.f) {
499       if (i > 1 && overlaps(this.ranges[iNumberOfRanges - i + 1], range)) {
500         mergeIfOverlaps(this.ranges[iNumberOfRanges - i + 1], range);
501       } else {
502           this.ranges.splice(iNumberOfRanges - i, 0, range);
503       }
504       return;
505     }
506   }
507   this.ranges.push(range);
508 };
509 /** The has method returns whether the specified number is include in the current rangeset instance
510   @param {Number} num The number to test for
511   @return {boolean} <code>true</code> if the number is in the range set (or absent and the inverted property is <code>true</code>), otherwise <code>false</code>.
512 */
513 RXBuild.Utils.RangeSet.prototype.has = function (num) {
514   var iNumberOfRanges = this.ranges.length;
515   for (var i=iNumberOfRanges; i > 0; i--) {
516     var oRange = this.ranges[iNumberOfRanges - i];
517     if (oRange.f <= num && oRange.t >= num)
518       return !this.inverted;
519     if (oRange.f > num) return this.inverted;
520   }
521   return this.inverted;
522 };
523 /** The toString method builds a string reprensentation of the current instances ranges
524   @return {string} A string representation of this instance (ex: <code>"1, 4..6, 7..9"</code>)
525 */
526 RXBuild.Utils.RangeSet.prototype.toString = function () {
527   var sResult = [];
528   var iNumberOfRanges = this.ranges.length;
529   for (var i=iNumberOfRanges; i > 0; i--) {
530     var oRange = this.ranges[iNumberOfRanges - i];
531     if (oRange.f == oRange.t)
532       sResult.push(oRange.f);
533     else
534       sResult.push(oRange.f + ".." + oRange.t);
535   }
536   return sResult.join(", ");
537 };
