 34 /** 
 35 	Creates a new instance of RXBuild.UI.RXEditor
 36 	@class The RXBuild.UI.RXEditor Is the main controller class for things related to editing a regular expression
 37 	@property {HTMLElement} container The html dom element that will parent the main pane of the editor
 38 	@property {HTMLElement} header The html dom element that will parent toolbar items of the editor
 39 	@property {HTMLElement} textBox The html dom element that is the actual pattern text editor
 40 	@property {HTMLElement} sampleInputTextBox The html dom element that lets the user edit the pattern input text
 41 	@property {YAHOO.util.CustomEvent} onRegExpChanged An event that is fired immediately when the user changes the regexp
 42 	@property {YAHOO.util.CustomEvent} onRegExpInvalidated An event that is fired immediately when the regexp is no longer valid
 43 	@property {YAHOO.util.CustomEvent} onRegExpApplyChanges An event that is fired when it is appropriate for an auto-refresh
 44 	@constructor
 45 	@param {String|HTMLElement} containerDiv The id or dom element that will parent the main pane of the editor
 46 	@param {String|HTMLElement} headerDiv The id or dom element that will parent toolbar items of the editor
 47 	@param {String|HTMLElement} textBox The id or html dom element that is the actual pattern text editor
 48 	@param {String|HTMLElement} sampleInputTextBox The id or html dom element that lets the user edit the pattern input text
 49 */
 50 RXBuild.UI.RXEditor = function(containerDiv, headerDiv, patternTextBox, inputTextBox) {
 51 	this.container = typeof(containerDiv) == "string" ? document.getElementById(containerDiv) : containerDiv;
 52 	this.header = typeof(headerDiv) == "string" ? document.getElementById(headerDiv) : headerDiv;
 53 	this.textBox = typeof(patternTextBox) == "string" ? document.getElementById(patternTextBox) : patternTextBox;
 54 	this.sampleInputTextBox = typeof(inputTextBox) == "string" ? document.getElementById(inputTextBox) : inputTextBox;
 55 	this.onRegExpChanged = new YAHOO.util.CustomEvent("onRegExpChanged");
 56 	this.onRegExpInvalidated = new YAHOO.util.CustomEvent("onRegExpInvalidated");
 57 	this.onRegExpApplyChanges = new YAHOO.util.CustomEvent("onRegExpUpdate");
 58 	this.changedEventTime = new RXBuild.UI.DelayedRefresh(300,
 59 		RXBuild.Utils.createDelegate(this, function() { this.onRegExpApplyChanges.fire(this.getRX()); }),
 60 		RXBuild.Utils.createDelegate(this, function() { this.onRegExpInvalidated.fire(this); }),
 61 		0.1);
 62 	YAHOO.util.Event.addListener(this.sampleInputTextBox, "keyup", RXBuild.Utils.createDelegate(this, function() {
 63 		this.__raiseRegExpChanged();
 64 	}));
 65 	
 66 };
 67 RXBuild.UI.RXEditor.prototype.constructor = RXBuild.UI.RXEditor;
 68 /** @private */
 69 RXBuild.UI.RXEditor.prototype.__raiseRegExpChanged = function() {
 70 	if (this.__previousRegexp) {
 71 		var oRX = this.getRX();
 72 		oRX.sampleInput = this.getInput();
 73 		if (this.__previousRegexp.options.str == oRX.options.str &&
 74 			this.__previousRegexp.pattern == oRX.pattern &&
 75 			this.__previousRegexp.sampleInput == oRX.sampleInput)
 76 			return; // Abort refresh.
 77 		this.__previousRegexp = oRX;
 78 	} else {
 79 		this.__previousRegexp = this.getRX();
 80 		this.__previousRegexp.sampleInput = this.getInput();
 81 	}
 82 	this.onRegExpChanged.fire(this);
 83 	this.changedEventTime.reset();
 84 };
 85 /** Reads the UI and returns an equivalent {@link RXBuild.RegExp} instance
 86 	@return {RXBuild.RegExp} A object describing the expression being edited
 87 */
 88 RXBuild.UI.RXEditor.prototype.getRX = function() {
 89 	return new RXBuild.RegExp(this.textBox.value, this.chkGlobal.checked, this.chkCaseInsensitive.checked, this.chkMultiline.checked);
 90 };

 96 };

104 /** Inserts the provided string into the regular expression at the current cursor position, overwriting the selection if any,
105 	and selecting the insertion (in some browsers apparently)
106 	@param {String} str The text to insert as-is into the regular expression
107 */
108 RXBuild.UI.RXEditor.prototype.insertIntoRegexp = function(str) {
109 	if (this.textBox == null) return;
110 	var iPos = this.__getCaretPosition();
111 	var sValueBefore = this.textBox.value.substr(0, iPos[0]);
112 	var sValueAfter = this.textBox.value.substr(iPos[1]);
113 	this.textBox.value = sValueBefore + str + sValueAfter;
114 	if (typeof(this.textBox.selectionStart) != "undefined") {
115 		this.textBox.selectionStart = sValueBefore.length;
116 		this.textBox.selectionEnd = sValueBefore.length + str.length;
117 	}
118 	this.__raiseRegExpChanged();
119 };
120 /** Returns the current selection in the pattern editor
121 	@private
122 	@return {Array} An array containing the cursors position followed by the index of the end of the selection.
123 */
124 RXBuild.UI.RXEditor.prototype.__getCaretPosition = function() {
125 	// thanks to http://blog.vishalon.net/index.php/javascript-getting-and-setting-caret-position-in-textarea/
126 	var iCaretPos = 0;
127 	var iEnd = 0;
128 	if (document.selection) { // IE & Opera Support
129 		this.textBox.focus();
130 		var Sel = document.selection.createRange();
131 		var iLen = Sel.text.length;
132 		Sel.moveStart('character', -this.textBox.value.length);
133 		iCaretPos = Sel.text.length;
134 		iEnd = iCaretPos;
135 		iCaretPos = iCaretPos - iLen;
136 	}
137 	else if (this.textBox.selectionStart || this.textBox.selectionStart == '0') {// Firefox support
138 		iCaretPos = this.textBox.selectionStart;
139 		iEnd = this.textBox.selectionEnd;
140 	}
141 	return [iCaretPos, iEnd];
142 };
143 /** Updates the caret position display div (positioning and content)
144 */
145 RXBuild.UI.RXEditor.prototype.updateCaretPositionDisplay = function () {
146 	var iPos = this.__getCaretPosition();
147 	this.divCaretPosition.setBody("c:" + iPos[0] + (iPos[1] != iPos[0] ? "-" + iPos[1] : ""));
148 	this.divCaretPosition.__ds_align();
149 	this.divCaretPosition.height = this.header.offsetHeight + "px";
150 };
151 RXBuild.UI.RXEditor.prototype.explainRegexp = function () {
152 	function makeHtmlCharsOfString(str) {
153 		var sResult = "";
154 		for (var i=0; i < str.length; i++) {
155 			sResult += "<span id=\"rxchr_" + i + "\">" + str.substr(i,1).escapeHTML() + "</span>";
156 		}
157 		return "<span class=\"input button\">" + sResult + "</span>";
158 	}
159 	var oRegExp = this.getRX();
160 	var oCompiledRegExp;
161 	try {
162 		oCompiledRegExp = oRegExp.compile();
163 		var sHTML = oCompiledRegExp.GetChainHtml();
164 		var oContainer = document.createElement("DIV");
165 		oContainer.className = "screenWide lightBox htmlDialog";
166 		oContainer.innerHTML = sHTML;
167 		oCompiledRegExp = oContainer;
168 	} catch (e) {
169 		oCompiledRegExp = e.plainTextToHtml();
170 	}
171 	var sRegexBlock = makeHtmlCharsOfString(oRegExp.pattern);
172 	RXBuild.UI.dlg.show("Explanation of " + sRegexBlock, oCompiledRegExp);
173 };
